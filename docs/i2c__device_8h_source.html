98  int read(uint8_t address, uint8_t* data, uint8_t size, BusConditions conditions = BusConditions::START_STOP)
99  {
100  bool ok = true;
101  if (uint8_t(conditions) & 0x01)
102  ok = (uint8_t(conditions) & 0x02 ? manager_.repeat_start() : manager_.start())
103  && manager_.send_slar(address);
104  while (ok && --size) ok = manager_.receive_data(*data++);
105  if (uint8_t(conditions) & 0x04)
106  {
107  ok = ok && manager_.receive_data(*data++, true);
108  manager_.stop();
109  }
110  return manager_.status();
111  }
