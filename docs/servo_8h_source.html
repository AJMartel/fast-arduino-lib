57  inline void set_pulse(uint16_t pulse_us)
58  {
59  // Constrain pulse to min/max and convert pulse to timer counter value
60  out_.set_duty(counter(utils::constrain(pulse_us, US_MINIMUM_, US_MAXIMUM_)));
61  }

63  inline void rotate(int8_t angle)
64  {
65  angle = utils::constrain(angle, MIN, MAX);
66  TYPE count =
67  (angle >= 0 ? utils::map(int32_t(angle), 0L, int32_t(MAX), COUNTER_NEUTRAL_, COUNTER_MAXIMUM_) :
68  utils::map(int32_t(angle), int32_t(MIN), 0L, COUNTER_MINIMUM_, COUNTER_NEUTRAL_));
69  out_.set_duty(count);
70  }
71 
72  private:
73  static constexpr TYPE counter(uint16_t pulse_us)
74  {
75  return CALC::PulseTimer_value(PRESCALER, pulse_us);
76  }
