138  // Conversion methods
139  static constexpr uint16_t echo_us_to_distance_mm(uint16_t echo_us)
140  {
141  // 340 m/s => 340000mm in 1000000us => 340/1000 mm/us
142  // Divide by 2 as echo time includes full sound round-trip
143  return uint16_t(echo_us * SPEED_OF_SOUND / 1000UL / 2UL);
144  }
145 
146  static constexpr uint16_t distance_mm_to_echo_us(uint16_t distance_mm)
147  {
148  // 340 m/s => 340000mm in 1000000us => 340/1000 mm/us
149  // Multiply by 2 as echo time must include full sound round-trip
150  return uint16_t(distance_mm * 1000UL * 2UL / SPEED_OF_SOUND);
151  }

160  template<board::Timer NTIMER_> class AbstractSonar
161  {
162  public:
163  using RTT = timer::RTT<NTIMER_>;
164 
165  inline bool ready() const
166  {
167  return status_ == READY;
168  }
169 
170  inline uint16_t latest_echo_us() const
171  {
172  synchronized
173  {
174  if (status_ == READY)
175  return echo_time_();
176  else
177  return 0;
178  }
179  }
180 
181  private:
182  using RAW_TIME = typename RTT::RAW_TIME;
183 
184  protected:
185  AbstractSonar(const RTT& rtt)
186  : rtt_{rtt}, status_{UNKNOWN}, timeout_time_ms_{},
187  echo_start_{RAW_TIME::EMPTY_TIME}, echo_end_{RAW_TIME::EMPTY_TIME}
188  {}
189 
190  uint16_t async_echo_us(uint16_t timeout_ms)
191  {
192  uint32_t now = rtt_.millis();
193  now += timeout_ms;
194  // Wait for echo signal start
195  while (status_ != READY)
196  if (rtt_.millis() >= now)
197  {
198  synchronized
199  {
200  status_ = READY;
201  echo_start_ = echo_end_ = RAW_TIME::EMPTY_TIME;
202  }
203  return 0;
204  }
205  return echo_time_();
206  }
207 
208  template<board::DigitalPin ECHO>
209  uint16_t blocking_echo_us(typename gpio::FastPinType<ECHO>::TYPE& echo, uint16_t timeout_ms)
210  {
211  uint32_t now = rtt_.millis();
212  now += timeout_ms;
213  while (!echo.value())
214  if (rtt_.millis() >= now) return 0;
215  synchronized
216  {
217  status_ = ECHO_STARTED;
218  echo_start_ = rtt_.raw_time();
219  }
220  // Wait for echo signal end
221  while (echo.value())
222  if (rtt_.millis() >= now) return 0;
223  synchronized
224  {
225  status_ = READY;
226  echo_end_ = rtt_.raw_time();
227  return echo_time_();
228  }
229  }
230 
231  inline void trigger_sent(uint16_t timeout_ms)
232  {
233  synchronized
234  {
235  status_ = TRIGGERED;
236  timeout_time_ms_ = rtt_.millis_() + timeout_ms;
237  }
238  }
239 
240  inline bool pulse_edge(bool rising)
241  {
242  if (rising && status_ == TRIGGERED)
243  {
244  status_ = ECHO_STARTED;
245  echo_start_ = rtt_.raw_time_();
246  }
247  else if ((!rising) && status_ == ECHO_STARTED)
248  {
249  status_ = READY;
250  echo_end_ = rtt_.raw_time_();
251  return true;
252  }
253  return false;
254  }
255 
256  inline bool rtt_time_changed()
257  {
258  if (status_ != READY && rtt_.millis_() >= timeout_time_ms_)
259  {
260  status_ = READY;
261  echo_start_ = echo_end_ = RAW_TIME::EMPTY_TIME;
262  return true;
263  }
264  return false;
265  }
266 
267  private:
268  uint16_t echo_time_() const
269  {
270  return uint16_t((echo_end_.as_real_time() - echo_start_.as_real_time()).total_micros());
271  }
272 
273  const RTT& rtt_;
274 
275  static constexpr const uint8_t UNKNOWN = 0x00;
276  static constexpr const uint8_t TRIGGERED = 0x10;
277  static constexpr const uint8_t ECHO_STARTED = 0x11;
278  static constexpr const uint8_t READY = 0x20;
279 
280  volatile uint8_t status_;
281  uint32_t timeout_time_ms_;
282 
283  RAW_TIME echo_start_;
284  RAW_TIME echo_end_;
285  };

287  template<board::Timer NTIMER_, board::DigitalPin TRIGGER_, board::DigitalPin ECHO_,
288  SonarType SONAR_TYPE_ = SonarType::BLOCKING>
289  class HCSR04 : public AbstractSonar<NTIMER_>
290  {
291  public:
292  using RTT = timer::RTT<NTIMER_>;
293  static constexpr const board::DigitalPin TRIGGER = TRIGGER_;
294  static constexpr const board::DigitalPin ECHO = ECHO_;
295  static constexpr const SonarType SONAR_TYPE = SONAR_TYPE_;
296 
297  private:
298  using PARENT = AbstractSonar<NTIMER_>;
299  using ECHO_PIN_TRAIT = board_traits::DigitalPin_trait<ECHO>;
300  using ECHO_PORT_TRAIT = board_traits::Port_trait<ECHO_PIN_TRAIT::PORT>;
301  static_assert(SONAR_TYPE != SonarType::ASYNC_INT || ECHO_PIN_TRAIT::IS_INT,
302  "SONAR_TYPE == ASYNC_INT but ECHO is not an INT pin");
303  static_assert(SONAR_TYPE != SonarType::ASYNC_PCINT || ECHO_PORT_TRAIT::PCINT != 0xFF,
304  "SONAR_TYPE == ASYNC_PCINT but ECHO is not an PCI pin");
305 
306  public:
307  static constexpr const uint16_t MAX_RANGE_M = 4;
308  static constexpr const uint16_t DEFAULT_TIMEOUT_MS = MAX_RANGE_M * 2 * 1000UL / SPEED_OF_SOUND + 1;
309 
310  HCSR04(const RTT& rtt) : PARENT{rtt}, trigger_{gpio::PinMode::OUTPUT}, echo_{gpio::PinMode::INPUT} {}
311 
312  inline void register_handler()
313  {
314  static_assert(SONAR_TYPE != SonarType::BLOCKING,
315  "register_handler() must not be called with SonarType::BLOCKING");
316  interrupt::register_handler(*this);
317  }
318 
319  // Blocking API
320  // Do note that timeout here is for the whole method not just for the sound echo, hence it
321  // must be bigger than just the time to echo the maximum roundtrip distance (typically x2)
322  uint16_t echo_us(uint16_t timeout_ms)
323  {
324  async_echo(timeout_ms);
325  return await_echo_us(timeout_ms);
326  }
327 
328  uint16_t await_echo_us(uint16_t timeout_ms)
329  {
330  if (SONAR_TYPE != SonarType::BLOCKING)
331  return this->async_echo_us(timeout_ms);
332  else
333  return this->template blocking_echo_us<ECHO>(echo_, timeout_ms);
334  }
335 
336  // We want to avoid using await_echo_us() to handle state & timeout!
337  void async_echo(uint16_t timeout_ms, bool trigger = true)
338  {
339  this->trigger_sent(timeout_ms);
340  if (trigger) this->trigger();
341  }
342 
343  private:
344  bool on_pin_change()
345  {
346  static_assert(SONAR_TYPE == SonarType::ASYNC_INT || SONAR_TYPE == SonarType::ASYNC_PCINT,
347  "on_pin_change() must be called only with SonarType::ASYNC_INT or ASYNC_PCINT");
348  return this->pulse_edge(echo_.value());
349  }
350 
351  bool on_rtt_change()
352  {
353  return this->rtt_time_changed();
354  }
355 
356  inline void trigger()
357  {
358  // Pulse TRIGGER for 10us
359  trigger_.set();
360  time::delay_us(TRIGGER_PULSE_US);
361  trigger_.clear();
362  }
363 
364  static constexpr const uint16_t TRIGGER_PULSE_US = 10;
365 
366  typename gpio::FastPinType<TRIGGER>::TYPE trigger_;
367  typename gpio::FastPinType<ECHO>::TYPE echo_;
368 
369  // Make friends with all ISR handlers
370  friend struct isr_handler;
371  };

407  template<board::Timer NTIMER_, board::DigitalPin TRIGGER_, board::Port ECHO_PORT_, uint8_t ECHO_MASK_>
408  class MultiHCSR04
409  {
410  public:
411  static constexpr const board::DigitalPin TRIGGER = TRIGGER_;
412  static constexpr const board::Port ECHO_PORT = ECHO_PORT_;
413  static constexpr const uint8_t ECHO_MASK = ECHO_MASK_;
414 
415  private:
416  using PTRAIT = board_traits::Port_trait<ECHO_PORT>;
417  static_assert(PTRAIT::PCINT != 0xFF, "ECHO_PORT_ must support PCINT");
418  static_assert((PTRAIT::DPIN_MASK & ECHO_MASK) == ECHO_MASK, "ECHO_MASK_ must contain available PORT pins");
419 
420  public:
421  using RTT = timer::RTT<NTIMER_>;
422  using EVENT = SonarEvent<NTIMER_>;
423 
424  static constexpr const uint16_t MAX_RANGE_M = 4;
425  static constexpr const uint16_t DEFAULT_TIMEOUT_MS = MAX_RANGE_M * 2 * 1000UL / SPEED_OF_SOUND + 1;
426 
427  MultiHCSR04(RTT& rtt)
428  : rtt_{rtt}, started_{}, ready_{}, active_{false},
429  timeout_time_ms_{}, trigger_{gpio::PinMode::OUTPUT}, echo_{0}
430  {
431  interrupt::register_handler(*this);
432  }
433 
434  uint8_t ready() const
435  {
436  return ready_;
437  }
438 
439  bool all_ready() const
440  {
441  return ready_ == ECHO_MASK;
442  }
443 
444  void set_ready()
445  {
446  if (active_)
447  {
448  ready_ = ECHO_MASK;
449  active_ = false;
450  }
451  }
452 
453  void trigger(uint16_t timeout_ms)
454  {
455  started_ = 0;
456  ready_ = 0;
457  timeout_time_ms_ = rtt_.millis() + timeout_ms;
458  active_ = true;
459  // Pulse TRIGGER for 10us
460  trigger_.set();
461  time::delay_us(TRIGGER_PULSE_US);
462  trigger_.clear();
463  }
464 
465  private:
466  EVENT on_pin_change()
467  {
468  if (!active_) return EVENT{};
469  // Compute the newly started echoes
470  uint8_t pins = echo_.get_PIN();
471  uint8_t started = pins & ~started_;
472  // Compute the newly finished echoes
473  uint8_t ready = ~pins & started_ & ~ready_;
474  // Update status of all echo pins
475  started_ |= started;
476  ready_ |= ready;
477  if (ready_ == ECHO_MASK) active_ = false;
478  return EVENT{started, ready, rtt_.raw_time_()};
479  }
480 
481  EVENT on_rtt_change()
482  {
483  if (active_ && rtt_.millis_() >= timeout_time_ms_)
484  {
485  active_ = false;
486  return EVENT{true};
487  }
488  return EVENT{};
489  }
490 
491  static constexpr const uint16_t TRIGGER_PULSE_US = 10;
492 
493  RTT& rtt_;
494  volatile uint8_t started_;
495  volatile uint8_t ready_;
496  volatile bool active_;
497  uint32_t timeout_time_ms_;
498  typename gpio::FastPinType<TRIGGER>::TYPE trigger_;
499  gpio::FastMaskedPort<ECHO_PORT, ECHO_MASK> echo_;
500 
501  // Make friends with all ISR handlers
502  friend struct isr_handler;
503  };
