504  template<board::Timer NTIMER_> class AbstractSonar
505  {
506  public:
508  using RTT = timer::RTT<NTIMER_>;
509 
519  inline bool ready() const
520  {
521  return status_ == READY;
522  }
523 
534  inline uint16_t latest_echo_us() const
535  {
536  synchronized
537  {
538  if (status_ == READY)
539  return echo_time_();
540  else
541  return 0;
542  }
543  }
544 
545  private:
546  using RAW_TIME = typename RTT::RAW_TIME;
547 
548  protected:
550  AbstractSonar(const RTT& rtt)
551  : rtt_{rtt}, status_{UNKNOWN}, timeout_time_ms_{},
552  echo_start_{RAW_TIME::EMPTY_TIME}, echo_end_{RAW_TIME::EMPTY_TIME}
553  {}
554 
555  uint16_t async_echo_us(uint16_t timeout_ms)
556  {
557  uint32_t now = rtt_.millis();
558  now += timeout_ms;
559  // Wait for echo signal start
560  while (status_ != READY)
561  if (rtt_.millis() >= now)
562  {
563  synchronized
564  {
565  status_ = READY;
566  echo_start_ = echo_end_ = RAW_TIME::EMPTY_TIME;
567  }
568  return 0;
569  }
570  return echo_time_();
571  }
572 
573  template<board::DigitalPin ECHO>
574  uint16_t blocking_echo_us(typename gpio::FastPinType<ECHO>::TYPE& echo, uint16_t timeout_ms)
575  {
576  uint32_t now = rtt_.millis();
577  now += timeout_ms;
578  while (!echo.value())
579  if (rtt_.millis() >= now) return 0;
580  synchronized
581  {
582  status_ = ECHO_STARTED;
583  echo_start_ = rtt_.raw_time();
584  }
585  // Wait for echo signal end
586  while (echo.value())
587  if (rtt_.millis() >= now) return 0;
588  synchronized
589  {
590  status_ = READY;
591  echo_end_ = rtt_.raw_time();
592  return echo_time_();
593  }
594  }
595 
596  inline void trigger_sent(uint16_t timeout_ms)
597  {
598  synchronized
599  {
600  status_ = TRIGGERED;
601  timeout_time_ms_ = rtt_.millis_() + timeout_ms;
602  }
603  }
604 
605  inline bool pulse_edge(bool rising)
606  {
607  if (rising && status_ == TRIGGERED)
608  {
609  status_ = ECHO_STARTED;
610  echo_start_ = rtt_.raw_time_();
611  }
612  else if ((!rising) && status_ == ECHO_STARTED)
613  {
614  status_ = READY;
615  echo_end_ = rtt_.raw_time_();
616  return true;
617  }
618  return false;
619  }
620 
621  inline bool rtt_time_changed()
622  {
623  if (status_ != READY && rtt_.millis_() >= timeout_time_ms_)
624  {
625  status_ = READY;
626  echo_start_ = echo_end_ = RAW_TIME::EMPTY_TIME;
627  return true;
628  }
629  return false;
630  }
632 
633  private:
634  uint16_t echo_time_() const
635  {
636  return uint16_t((echo_end_.as_real_time() - echo_start_.as_real_time()).total_micros());
637  }
638 
639  const RTT& rtt_;
640 
641  static constexpr const uint8_t UNKNOWN = 0x00;
642  static constexpr const uint8_t TRIGGERED = 0x10;
643  static constexpr const uint8_t ECHO_STARTED = 0x11;
644  static constexpr const uint8_t READY = 0x20;
645 
646  volatile uint8_t status_;
647  uint32_t timeout_time_ms_;
648 
649  RAW_TIME echo_start_;
650  RAW_TIME echo_end_;
651  };

1099  uint8_t started = pins & ~started_;
1100  // Compute the newly finished echoes
