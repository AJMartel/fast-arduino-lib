201  static constexpr uint16_t echo_us_to_distance_mm(uint16_t echo_us)

203  // 340 m/s => 340000mm in 1000000us => 340/1000 mm/us

205  return uint16_t(echo_us * SPEED_OF_SOUND / 1000UL / 2UL);

210  // 340 m/s => 340000mm in 1000000us => 340/1000 mm/us
211  // Multiply by 2 as echo time must include full sound round-trip
212  return uint16_t(distance_mm * 1000UL * 2UL / SPEED_OF_SOUND);

269  TYPE async_echo_ticks(TYPE timeout_ticks)
270  {
271  // Wait for echo signal start
272  while (!(status_ & READY))
273  if (timer_.ticks() >= timeout_ticks)
274  {
275  status_ = READY;
276  return 0;
277  }
278  return echo_pulse_;
279  }
280 
281  template<board::DigitalPin ECHO>
282  TYPE blocking_echo_ticks(typename gpio::FastPinType<ECHO>::TYPE& echo, TYPE timeout_ticks)
283  {
284  timer_.reset();
285  while (!echo.value())
286  if (timer_.ticks() >= timeout_ticks)
287  return 0;
288  echo_pulse_ = timer_.ticks();
289  status_ = STARTED;
290  // Wait for echo signal end
291  while (echo.value())
292  if (timer_.ticks() >= timeout_ticks)
293  return 0;
294  echo_pulse_ = timer_.ticks() - echo_pulse_;
295  status_ = READY;
296  return echo_pulse_;
297  }
298 
299  inline void trigger_sent(bool reset)
300  {
301  // Trigger capture if needed (compile-time decision)
302  TimerTrigger<CAPTURE>::trigger(timer_);
303  if (reset) timer_.reset();
304  status_ = 0;
305  }
306 
307  inline bool pulse_edge(bool rising, TYPE ticks)
308  {
309  if (rising && status_ == 0)
310  {
311  echo_pulse_ = ticks;
312  status_ = STARTED;
313  }
314  else if ((!rising) && status_ == STARTED)
315  {
316  echo_pulse_ = ticks - echo_pulse_;
317  status_ = READY;
318  return true;
319  }
320  return false;
321  }
322 
323  inline bool pulse_captured(TYPE capture)
324  {
325  //TODO static_assert
326  bool rising = (timer_.input_capture() == timer::TimerInputCapture::RISING_EDGE);
327  if (rising)
328  timer_.set_input_capture(timer::TimerInputCapture::FALLING_EDGE);
329  return pulse_edge(rising, capture);
330  }

334  volatile TYPE echo_pulse_;

336  static constexpr const uint8_t READY = 0x01;
337  static constexpr const uint8_t STARTED = 0x02;
338  volatile uint8_t status_;

511  {
512  started_ = 0;
513  ready_ = 0;
514  active_ = true;
515  timer_.reset();
516  // Pulse TRIGGER for 10us
517  trigger_.set();
518  time::delay_us(TRIGGER_PULSE_US);
519  trigger_.clear();
520  }
521 
522  EVENT on_pin_change()
523  {
524  //TODO Better get ticks from ISR and pass it along as argument
525  TYPE ticks = timer_.ticks_();
526  if (!active_)
527  return EVENT{};
528  // Compute the newly started echoes
529  uint8_t pins = echo_.get_PIN();
530  uint8_t started = pins & ~started_;
531  // Compute the newly finished echoes
532  uint8_t ready = ~pins & started_ & ~ready_;
533  // Update status of all echo pins
534  started_ |= started;
535  ready_ |= ready;
536  if (ready_ == ECHO_MASK)
537  active_ = false;
538  return EVENT{started, ready, ticks};
539  }
