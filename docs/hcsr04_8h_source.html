101  static constexpr const uint32_t SPEED_OF_SOUND = 340UL;
102 
103  // Conversion methods
104  static constexpr uint16_t echo_us_to_distance_mm(uint16_t echo_us)
105  {
106  // 340 m/s => 340000mm in 1000000us => 340/1000 mm/us
107  // Divide by 2 as echo time includes full sound round-trip
108  return uint16_t(echo_us * SPEED_OF_SOUND / 1000UL / 2UL);
109  }
110 
111  static constexpr uint16_t distance_mm_to_echo_us(uint16_t distance_mm)
112  {
113  // 340 m/s => 340000mm in 1000000us => 340/1000 mm/us
114  // Multiply by 2 as echo time must include full sound round-trip
115  return uint16_t(distance_mm * 1000UL * 2UL / SPEED_OF_SOUND);
116  }

118  //TODO improve API to allow several types instances sharing the same TRIGGER pin
119  template<board::Timer TIMER, board::DigitalPin TRIGGER, board::DigitalPin ECHO>
120  class HCSR04
121  {
122  public:
123  static constexpr const uint16_t MAX_RANGE_M = 4;
124  static constexpr const uint16_t DEFAULT_TIMEOUT_MS = MAX_RANGE_M * 2 * 1000 / SPEED_OF_SOUND + 1;
125 
126  HCSR04(timer::RTT<TIMER>& rtt, bool async = true)
127  : rtt_{rtt},
128  trigger_{gpio::PinMode::OUTPUT}, echo_{gpio::PinMode::INPUT},
129  start_{}, echo_pulse_{0}, status_{0}
130  {
131  // If async mode is gonna be used, then register this as interrupt handler (for INT or PCI)
132  if (async)
133  interrupt::register_handler(*this);
134  }
135 
136  // Blocking API
137  // Do note that timeout here is for the whole method not just for the sound echo, hence it
138  // must be bigger than just the time to echo the maximum roundtrip distance (typically x2)
139  uint16_t echo_us(uint16_t timeout_ms = DEFAULT_TIMEOUT_MS)
140  {
141  trigger();
142  // Wait for echo signal start
143  timeout_ms += rtt_.millis();
144  while (!echo_.value())
145  if (rtt_.millis() >= timeout_ms) return 0;
146  // Read current time (need RTT)
147  time::RTTTime start = rtt_.time();
148  // Wait for echo signal end
149  while (echo_.value())
150  if (rtt_.millis() >= timeout_ms) return 0;
151  // Read current time (need RTT)
152  time::RTTTime end = rtt_.time();
153  time::RTTTime delta = time::delta(start, end);
154  echo_pulse_ = uint16_t(delta.millis * 1000UL + delta.micros);
155  return echo_pulse_;
156  }
157 
158  //TODO How to manage status & timeout smartly?
159  // We want to avoid using await_echo_us() to handle state & timeout!
160  void async_echo(bool trigger = true)
161  {
162  status_ = 0;
163  if (trigger) this->trigger();
164  }
165 
166  bool ready() const
167  {
168  return status_ & READY;
169  }
170 
171  uint16_t latest_echo_us() const
172  {
173  synchronized return echo_pulse_;
174  }
175 
176  uint16_t await_echo_us(uint16_t timeout_ms = DEFAULT_TIMEOUT_MS)
177  {
178  // Wait for echo signal start
179  timeout_ms += rtt_.millis();
180  while (!(status_ & READY))
181  if (rtt_.millis() >= timeout_ms)
182  {
183  status_ = READY;
184  return 0;
185  }
186  return echo_pulse_;
187  }
188 
189  void on_echo()
190  {
191  if (echo_.value())
192  {
193  // pulse started
194  start_ = rtt_.time();
195  status_ = STARTED;
196  }
197  else if (status_ & STARTED)
198  {
199  // pulse ended
200  time::RTTTime end = rtt_.time();
201  time::RTTTime delta = time::delta(start_, end);
202  echo_pulse_ = uint16_t(delta.millis * 1000UL + delta.micros);
203  status_ = READY;
204  }
205  }
206 
207  private:
208  void trigger()
209  {
210  rtt_.millis(0);
211  // Pulse TRIGGER for 10us
212  trigger_.set();
213  time::delay_us(TRIGGER_PULSE_US);
214  trigger_.clear();
215  }
216 
217  static constexpr const uint16_t TRIGGER_PULSE_US = 10;
218 
219  timer::RTT<TIMER>& rtt_;
220  typename gpio::FastPinType<TRIGGER>::TYPE trigger_;
221  typename gpio::FastPinType<ECHO>::TYPE echo_;
222  time::RTTTime start_;
223  volatile uint16_t echo_pulse_;
224 
225  static constexpr const uint8_t READY = 0x01;
226  static constexpr const uint8_t STARTED = 0x02;
227  volatile uint8_t status_;
228  };
229 }
230 }
