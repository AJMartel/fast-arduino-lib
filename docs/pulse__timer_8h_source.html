28  ISR(CAT3(TIMER, TIMER_NUM, _OVF_vect)) \
29  { \
30  const board::Timer T = CAT(board::Timer::TIMER, TIMER_NUM); \
31  using TT = board_traits::Timer_trait<T>; \
32  static_assert(!TT::IS_16BITS, "TIMER_NUM must be an 8 bits Timer"); \
33  using PINTA = board_traits::Timer_COM_trait<T, COM_A>; \
34  static_assert(PIN_A == PINTA::PIN_OCR, "PIN_A must be connected to TIMER_NUM OCxA"); \
35  using PINTB = board_traits::Timer_COM_trait<T, COM_B>; \
36  static_assert(PIN_B == PINTB::PIN_OCR, "PIN_B must be connected to TIMER_NUM OCxB"); \
37  using PT = timer::PulseTimer8<T, PRESCALER>; \
38  if (CALL_HANDLER_RETURN_(PT, &PT::overflow, bool)()) \
39  { \
40  if (PINTA::OCR != 0) gpio::FastPinType<PIN_A>::set(); \
41  if (PINTB::OCR != 0) gpio::FastPinType<PIN_B>::set(); \
42  } \
43  }

85  // Timer specialized in emitting pulses with accurate width, according to a slow frequency; this is typically
86  // useful for controlling servos, which need a pulse with a width range from ~1000us to ~2000us, send every
87  // 20ms, ie with a 50Hz frequency.
88  // This implementation ensures a good pulse width precision for 16-bits timer.
89  template<board::Timer NTIMER_, typename Calculator<NTIMER_>::PRESCALER PRESCALER_>
90  class PulseTimer16 : public Timer<NTIMER_>
91  {
92  public:
93  static constexpr const board::Timer NTIMER = NTIMER_;
94 
95  private:
96  using PARENT = Timer<NTIMER>;
97  using TRAIT = typename PARENT::TRAIT;
98  static_assert(TRAIT::IS_16BITS, "TIMER must be a 16 bits timer");
99 
100  public:
101  using CALCULATOR = Calculator<NTIMER>;
102  using TPRESCALER = typename CALCULATOR::PRESCALER;
103  static constexpr const TPRESCALER PRESCALER = PRESCALER_;
104 
105  PulseTimer16(uint16_t pulse_frequency) : Timer<NTIMER>{TCCRA_MASK(), TCCRB_MASK()}
106  {
107  TRAIT::ICR = CALCULATOR::PWM_ICR_counter(PRESCALER, pulse_frequency);
108  }
109 
110  private:
111  static constexpr uint8_t TCCRA_MASK()
112  {
113  // If 16 bits, use ICR1 FastPWM
114  return TRAIT::F_PWM_ICR_TCCRA;
115  }
116  static constexpr uint8_t TCCRB_MASK()
117  {
118  // If 16 bits, use ICR1 FastPWM and prescaler forced to best fit all pulse frequency
119  return TRAIT::F_PWM_ICR_TCCRB | TRAIT::TCCRB_prescaler(PRESCALER);
120  }
121  };

123  // Timer specialized in emitting pulses with accurate width, according to a slow frequency; this is typically
124  // useful for controlling servos, which need a pulse with a width range from ~1000us to ~2000us, send every
125  // 20ms, ie with a 50Hz frequency.
126  // This implementation ensures a good pulse width precision for 8-bits timers.
127  template<board::Timer NTIMER_, typename Calculator<NTIMER_>::PRESCALER PRESCALER_>
128  class PulseTimer8 : public Timer<NTIMER_>
129  {
130  public:
131  static constexpr const board::Timer NTIMER = NTIMER_;
132 
133  private:
134  using PARENT = Timer<NTIMER>;
135  using TRAIT = typename PARENT::TRAIT;
136  static_assert(!TRAIT::IS_16BITS, "TIMER must be an 8 bits timer");
137 
138  public:
139  using CALCULATOR = Calculator<NTIMER>;
140  using TPRESCALER = typename CALCULATOR::PRESCALER;
141  static constexpr const TPRESCALER PRESCALER = PRESCALER_;
142 
143  PulseTimer8(uint16_t pulse_frequency)
144  : Timer<NTIMER>{TCCRA_MASK(), TCCRB_MASK(), TIMSK_INT_MASK()}, MAX{OVERFLOW_COUNTER(pulse_frequency)}
145  {
146  // If 8 bits timer, then we need ISR on Overflow and Compare A/B
147  interrupt::register_handler(*this);
148  }
149 
150  bool overflow()
151  {
152  if (++count_ == MAX) count_ = 0;
153  return !count_;
154  }
155 
156  private:
157  static constexpr uint8_t TCCRA_MASK()
158  {
159  // If 8 bits, use CTC/TOV ISR
160  return 0;
161  }
162  static constexpr uint8_t TCCRB_MASK()
163  {
164  // If 8 bits, use CTC/TOV ISR with prescaler forced best fit max pulse width
165  return TRAIT::TCCRB_prescaler(PRESCALER);
166  }
167  static constexpr uint8_t TIMSK_INT_MASK()
168  {
169  return TRAIT::TIMSK_INT_MASK(uint8_t(TimerInterrupt::OVERFLOW | TimerInterrupt::OUTPUT_COMPARE_A |
170  TimerInterrupt::OUTPUT_COMPARE_B));
171  }
172  static constexpr uint8_t OVERFLOW_COUNTER(uint16_t pulse_frequency)
173  {
174  return F_CPU / 256UL / _BV(uint8_t(PRESCALER)) / pulse_frequency;
175  }
176 
177  private:
178  const uint8_t MAX;
179  uint8_t count_;
180  };
