156  bool begin( GyroRange gyro_range = GyroRange::RANGE_250,
157  AccelRange accel_range = AccelRange::RANGE_2G,
158  DLPF low_pass_filter = DLPF::ACCEL_BW_260HZ,
159  ClockSelect clock_select = ClockSelect::INTERNAL_8MHZ)
160  {
161  PowerManagement power;
162  power.clock_select = uint8_t(clock_select);
163  return this->write(DEVICE_ADDRESS, CONFIG, i2c::BusConditions::START_NO_STOP) == i2c::Status::OK
164  && this->write(DEVICE_ADDRESS, uint8_t(low_pass_filter), i2c::BusConditions::NO_START_NO_STOP) == i2c::Status::OK
165  && this->write(DEVICE_ADDRESS, uint8_t(gyro_range), i2c::BusConditions::NO_START_NO_STOP) == i2c::Status::OK
166  && this->write(DEVICE_ADDRESS, uint8_t(accel_range), i2c::BusConditions::NO_START_NO_STOP) == i2c::Status::OK
167  && this->write(DEVICE_ADDRESS, PWR_MGMT_1, i2c::BusConditions::REPEAT_START_NO_STOP) == i2c::Status::OK
168  && this->write(DEVICE_ADDRESS, power, i2c::BusConditions::NO_START_STOP) == i2c::Status::OK;
169  }

171  bool begin( FIFOEnable fifo_enable,
172  INTStatus int_enable,
173  uint8_t sample_rate_divider,
174  GyroRange gyro_range = GyroRange::RANGE_250,
175  AccelRange accel_range = AccelRange::RANGE_2G,
176  DLPF low_pass_filter = DLPF::ACCEL_BW_260HZ,
177  ClockSelect clock_select = ClockSelect::INTERNAL_8MHZ)
178  {
179  fifo_enable.reserved = 0;
180  if (utils::as_uint8_t(fifo_enable) == 0)
181  return begin(gyro_range, accel_range, low_pass_filter, clock_select);
182  int_enable.reserved1 = int_enable.reserved2 = 0;
183  PowerManagement power;
184  power.clock_select = uint8_t(clock_select);
185  return this->write(DEVICE_ADDRESS, CONFIG, i2c::BusConditions::START_NO_STOP) == i2c::Status::OK
186  && this->write(DEVICE_ADDRESS, uint8_t(low_pass_filter), i2c::BusConditions::NO_START_NO_STOP) == i2c::Status::OK
187  && this->write(DEVICE_ADDRESS, uint8_t(gyro_range), i2c::BusConditions::NO_START_NO_STOP) == i2c::Status::OK
188  && this->write(DEVICE_ADDRESS, uint8_t(accel_range), i2c::BusConditions::NO_START_NO_STOP) == i2c::Status::OK
189 
190  && this->write(DEVICE_ADDRESS, SMPRT_DIV, i2c::BusConditions::REPEAT_START_NO_STOP) == i2c::Status::OK
191  && this->write(DEVICE_ADDRESS, sample_rate_divider, i2c::BusConditions::NO_START_NO_STOP) == i2c::Status::OK
192 
193  && this->write(DEVICE_ADDRESS, FIFO_EN, i2c::BusConditions::REPEAT_START_NO_STOP) == i2c::Status::OK
194  && this->write(DEVICE_ADDRESS, utils::as_uint8_t(fifo_enable), i2c::BusConditions::NO_START_NO_STOP) == i2c::Status::OK
195 
196  && this->write(DEVICE_ADDRESS, INT_PIN_CFG, i2c::BusConditions::REPEAT_START_NO_STOP) == i2c::Status::OK
197  && this->write(DEVICE_ADDRESS, uint8_t(0), i2c::BusConditions::NO_START_NO_STOP) == i2c::Status::OK
198  && this->write(DEVICE_ADDRESS, utils::as_uint8_t(int_enable), i2c::BusConditions::NO_START_NO_STOP) == i2c::Status::OK
199 
200  && this->write(DEVICE_ADDRESS, USER_CTRL, i2c::BusConditions::REPEAT_START_NO_STOP) == i2c::Status::OK
201  && this->write(DEVICE_ADDRESS, uint8_t(0x40), i2c::BusConditions::NO_START_NO_STOP) == i2c::Status::OK
202  && this->write(DEVICE_ADDRESS, power, i2c::BusConditions::NO_START_STOP) == i2c::Status::OK;
203  }

340  bool fifo_pop(uint8_t* buffer, uint8_t size, bool wait)
341  {
342  while (fifo_count() < size)
343  if (!wait)
344  return false;
345  else
346  {
347  //TODO yield here instead?
348  }
349  if ( this->write(DEVICE_ADDRESS, FIFO_R_W, i2c::BusConditions::START_NO_STOP) == i2c::Status::OK
350  && this->read(DEVICE_ADDRESS, buffer, size, i2c::BusConditions::REPEAT_START_STOP) == i2c::Status::OK)
351  {
352  // Swap all 2-bytes words
353  uint16_t* temp = (uint16_t*) buffer;
354  size /= 2;
355  while (size--) utils::swap_bytes(*temp++);
356  return true;
357  }
358  return false;
359  }
