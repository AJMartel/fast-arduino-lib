293  bool begin( FIFOEnable fifo_enable,
294  INTEnable int_enable,
295  uint8_t sample_rate_divider,
296  GyroRange gyro_range = GyroRange::RANGE_250,
297  AccelRange accel_range = AccelRange::RANGE_2G,

299  ClockSelect clock_select = ClockSelect::INTERNAL_8MHZ)

303  if (utils::as_uint8_t(fifo_enable) == 0)
304  return begin(gyro_range, accel_range, low_pass_filter, clock_select);
305  int_enable.reserved1 = int_enable.reserved2 = 0;
306  PowerManagement power;

308  return this->write(DEVICE_ADDRESS, CONFIG, BusCond::START_NO_STOP) == OK
309  && this->write(DEVICE_ADDRESS, uint8_t(low_pass_filter), BusCond::NO_START_NO_STOP) == OK

312

314  && this->write(DEVICE_ADDRESS, sample_rate_divider, BusCond::NO_START_NO_STOP) == OK
315 
316  && this->write(DEVICE_ADDRESS, FIFO_EN, BusCond::REPEAT_START_NO_STOP) == OK
317  && this->write(DEVICE_ADDRESS, utils::as_uint8_t(fifo_enable), BusCond::NO_START_NO_STOP) == OK

586  bool fifo_pop(uint8_t* buffer, uint8_t size, bool wait)
587  {
588  using namespace i2c::Status;
589  while (fifo_count() < size)
590  if (!wait)
591  return false;
592  else
593  {
594  //TODO yield here instead?
595  }
596  if (this->write(DEVICE_ADDRESS, FIFO_R_W, BusCond::START_NO_STOP) == OK
597  && this->read(DEVICE_ADDRESS, buffer, size, BusCond::REPEAT_START_STOP) == OK)
598  {
599  // Swap all 2-bytes words
600  uint16_t* temp = (uint16_t*) buffer;
601  size /= 2;
602  while (size--) utils::swap_bytes(*temp++);
603  return true;
604  }
605  return false;
606  }
