112  template<board::Timer NTIMER, board::DigitalPin OUTPUT> class ToneGenerator
113  {
114  private:
115  using SQWGEN = timer::SquareWave<NTIMER, OUTPUT>;
116  static constexpr const uint32_t INTERTONE_DELAY_MS = 20;
117 
118  public:
119  using PRESCALER = typename SQWGEN::TIMER::PRESCALER;
120  using COUNTER = typename SQWGEN::TIMER::TYPE;
121 
122  ToneGenerator() : generator_{} {}
123 
124  void start_tone(Tone t)
125  {
126  if (t > Tone::SILENCE) generator_.start_frequency(uint32_t(t));
127  }
128  inline void start_tone(PRESCALER prescaler, COUNTER counter)
129  {
130  generator_.start_frequency(prescaler, counter);
131  }
132  inline void stop_tone()
133  {
134  generator_.stop();
135  }
136 
137  void tone(Tone t, uint16_t ms)
138  {
139  if (t > Tone::SILENCE) generator_.start_frequency(uint32_t(t));
140  if (t >= Tone::SILENCE) pause(ms);
141  }
142  inline void tone(PRESCALER prescaler, COUNTER counter, uint16_t ms)
143  {
144  generator_.start_frequency(prescaler, counter);
145  pause(ms);
146  }
147  inline void pause(uint16_t ms)
148  {
149  time::delay_ms(ms);
150  generator_.stop();
151  // Short delay between tones
152  time::delay_ms(INTERTONE_DELAY_MS);
153  }
154 
155  private:
156  SQWGEN generator_;
157  };
