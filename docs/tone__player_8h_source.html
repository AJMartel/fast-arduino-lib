333  void play_(const TonePlay* melody, LOAD_TONE load_tone)
334  {
335  const TonePlay* repeat_play = 0;
336  int8_t repeat_times;
337  const TonePlay* play = melody;
338  while (true)
339  {
340  TonePlay holder;
341  const TonePlay* current = load_tone(play, holder);
342  if (current->tone == SpecialTone::END) break;
343  if (current->tone == SpecialTone::REPEAT_START)
344  {
345  repeat_play = play;
346  repeat_times = -1;
347  }
348  else if (current->tone == SpecialTone::REPEAT_END)
349  {
350  if (repeat_play != 0)
351  {
352  if (repeat_times == -1) repeat_times = current->ms;
353  if (repeat_times--)
354  play = repeat_play;
355  else
356  repeat_play = 0;
357  }
358  }
359  else
360  generator_.tone(current->tone, current->ms);
361  ++play;
362  }
363  }

382  void play_(const QTonePlay* melody, LOAD_QTONE load_tone)
383  {
384  const QTonePlay* repeat_play = 0;
385  int8_t repeat_times;
386  const QTonePlay* play = melody;
387  while (true)
388  {
389  QTonePlay holder;
390  const QTonePlay* current = load_tone(play, holder);
391  if (current->is_end()) break;
392  if (current->is_repeat_start())
393  {
394  repeat_play = play;
395  repeat_times = -1;
396  }
397  else if (current->is_repeat_end())
398  {
399  if (repeat_play != 0)
400  {
401  if (repeat_times == -1) repeat_times = current->repeat_count();
402  if (repeat_times--)
403  play = repeat_play;
404  else
405  repeat_play = 0;
406  }
407  }
408  else if (current->is_pause())
409  generator_.pause(current->duration());
410  else
411  generator_.tone(current->prescaler(), current->counter(), current->duration());
412  ++play;
413  }
414  }
