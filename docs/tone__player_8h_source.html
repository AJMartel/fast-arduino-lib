106  static constexpr uint32_t period(Tone tone)
107  {
108  return 1000000UL / 2 / uint16_t(tone);
109  }
110  static constexpr PRESCALER prescaler(Tone tone)
111  {
112  return (tone > Tone::SILENCE ? CALC::CTC_prescaler(period(tone)) : PRESCALER::NO_PRESCALING);
113  }
114  static constexpr COUNTER counter(Tone tone)
115  {
116  return (tone > Tone::SILENCE ? CALC::CTC_counter(prescaler(tone), period(tone)) : 0);
117  }

173  void play_(const TonePlay* melody, LOAD_TONE load_tone)
174  {
175  const TonePlay* repeat_play = 0;
176  int8_t repeat_times;
177  const TonePlay* play = melody;
178  while (true)
179  {
180  TonePlay holder;
181  const TonePlay* current = load_tone(play, holder);
182  if (current->tone == SpecialTone::END) break;
183  if (current->tone == SpecialTone::REPEAT_START)
184  {
185  repeat_play = play;
186  repeat_times = -1;
187  }
188  else if (current->tone == SpecialTone::REPEAT_END)
189  {
190  if (repeat_play != 0)
191  {
192  if (repeat_times == -1) repeat_times = current->ms;
193  if (repeat_times--)
194  play = repeat_play;
195  else
196  repeat_play = 0;
197  }
198  }
199  else
200  generator_.tone(current->tone, current->ms);
201  ++play;
202  }
203  }

222  void play_(const QTonePlay* melody, LOAD_QTONE load_tone)
223  {
224  const QTonePlay* repeat_play = 0;
225  int8_t repeat_times;
226  const QTonePlay* play = melody;
227  while (true)
228  {
229  QTonePlay holder;
230  const QTonePlay* current = load_tone(play, holder);
231  if (current->is_end()) break;
232  if (current->is_repeat_start())
233  {
234  repeat_play = play;
235  repeat_times = -1;
236  }
237  else if (current->is_repeat_end())
238  {
239  if (repeat_play != 0)
240  {
241  if (repeat_times == -1) repeat_times = current->repeat_count();
242  if (repeat_times--)
243  play = repeat_play;
244  else
245  repeat_play = 0;
246  }
247  }
248  else if (current->is_pause())
249  generator_.pause(current->duration());
250  else
251  generator_.tone(current->prescaler(), current->counter(), current->duration());
252  ++play;
253  }
254  }
255 
256  GENERATOR& generator_;
257  };
