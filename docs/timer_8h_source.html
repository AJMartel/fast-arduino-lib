561  static constexpr TIMER_PRESCALER best_prescaler_in_2(TIMER_PRESCALER p1, TIMER_PRESCALER p2, uint32_t us)
562  {
563  return (!prescaler_is_adequate(prescaler_quotient(p1, us)) ? p2 :
564  !prescaler_is_adequate(prescaler_quotient(p2, us)) ? p1 :
565  prescaler_remainder(p1, us) < prescaler_remainder(p2, us) ? p1 :
566  prescaler_remainder(p1, us) > prescaler_remainder(p2, us) ? p2 :
567  prescaler_quotient(p1, us) > prescaler_quotient(p2, us) ? p1 : p2);
568  }
