571  static constexpr PRESCALER best_prescaler_in_2(PRESCALER p1, PRESCALER p2, uint32_t us)
572  {
573  return (!prescaler_is_adequate(prescaler_quotient(p1, us)) ?
574  p2 :
575  !prescaler_is_adequate(prescaler_quotient(p2, us)) ?
576  p1 :
577  prescaler_remainder(p1, us) < prescaler_remainder(p2, us) ?
578  p1 :
579  prescaler_remainder(p1, us) > prescaler_remainder(p2, us) ?
580  p2 :
581  prescaler_quotient(p1, us) > prescaler_quotient(p2, us) ? p1 : p2);
582  }

585  {
586  return (begin + 1 == end ? *begin : best_prescaler_in_2(*begin, best_prescaler(begin + 1, end, us), us));
587  }
