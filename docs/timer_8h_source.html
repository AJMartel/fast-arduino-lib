433  static constexpr TIMER_PRESCALER best_prescaler_in_2(TIMER_PRESCALER p1, TIMER_PRESCALER p2, uint32_t us)
434  {
435  return (!prescaler_is_adequate(prescaler_quotient(p1, us)) ? p2 :
436  !prescaler_is_adequate(prescaler_quotient(p2, us)) ? p1 :
437  prescaler_remainder(p1, us) < prescaler_remainder(p2, us) ? p1 :
438  prescaler_remainder(p1, us) > prescaler_remainder(p2, us) ? p2 :
439  prescaler_quotient(p1, us) > prescaler_quotient(p2, us) ? p1 : p2);
440  }

443  {
444  return (begin + 1 == end ? *begin : best_prescaler_in_2(*begin, best_prescaler(begin + 1 , end, us), us));
445  }
