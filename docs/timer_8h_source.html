594  static constexpr PRESCALER best_prescaler_in_2(PRESCALER p1, PRESCALER p2, uint32_t us)
595  {
596  return (!prescaler_is_adequate(prescaler_quotient(p1, us)) ?
597  p2 :
598  !prescaler_is_adequate(prescaler_quotient(p2, us)) ?
599  p1 :
600  prescaler_remainder(p1, us) < prescaler_remainder(p2, us) ?
601  p1 :
602  prescaler_remainder(p1, us) > prescaler_remainder(p2, us) ?
603  p2 :
604  prescaler_quotient(p1, us) > prescaler_quotient(p2, us) ? p1 : p2);
605  }

610  }
