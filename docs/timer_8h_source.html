562  static constexpr TIMER_PRESCALER best_prescaler_in_2(TIMER_PRESCALER p1, TIMER_PRESCALER p2, uint32_t us)
563  {
564  return (!prescaler_is_adequate(prescaler_quotient(p1, us)) ? p2 :
565  !prescaler_is_adequate(prescaler_quotient(p2, us)) ? p1 :
566  prescaler_remainder(p1, us) < prescaler_remainder(p2, us) ? p1 :
567  prescaler_remainder(p1, us) > prescaler_remainder(p2, us) ? p2 :
568  prescaler_quotient(p1, us) > prescaler_quotient(p2, us) ? p1 : p2);
569  }

572  {
573  return (begin + 1 == end ? *begin : best_prescaler_in_2(*begin, best_prescaler(begin + 1 , end, us), us));
574  }
