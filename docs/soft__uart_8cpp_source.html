19 void serial::soft::AbstractUATX::begin_serial(uint32_t rate, Parity parity, StopBits stop_bits)
20 {
21  parity_ = parity;
22  // Calculate timing for TX in number of cycles
23  uint16_t bit_time = uint16_t(F_CPU / rate);
24  // Actual timing is based on number of times to count 4 cycles, because we use _delay_loop_2()
25  // 11 or 12 cycles + delay counted from start bit (cbi) to first bit (sbi or cbi)
26  interbit_tx_time_ = (bit_time - 12) / 4;
27  // 11 or 12 cycles + delay counted from first bit (sbi or cbi) to second bit (sbi or cbi)
28  start_bit_tx_time_ = (bit_time - 12) / 4;
29  // For stop bit we lengthten the bit duration of 25% to guarantee alignment of RX side on stop duration
30  stop_bit_tx_time_ = (bit_time / 4) * 5 / 4;
31  if (stop_bits == StopBits::TWO) stop_bit_tx_time_ *= 2;
32 }

34 static constexpr uint16_t compute_delay(uint16_t total_cycles, uint16_t less_cycles)
35 {
36  // We add 3 cycles to allow rounding
37  return (total_cycles > less_cycles ? (total_cycles - less_cycles + 3) / 4 : 1);

40 void serial::soft::AbstractUARX::begin_serial(uint32_t rate, Parity parity, UNUSED StopBits stop_bits)
41 {
42  parity_ = parity;
43  // Calculate timing for RX in number of cycles
44  uint16_t bit_time = uint16_t(F_CPU / rate);
45 
46  // Actual timing is based on number of times to count 4 cycles, because we use _delay_loop_2()
47  // 87 cycles (+4N delay) elapse until start bit is detected from PCI interrupt and 1st bit is sampled:
48  // - 3 cycles to generate the PCI interrupt
49  // - 1-4 (take 2) cycles to complete current instruction
50  // - 4 cycles to process the interrupt + 2 cycles rjmp in vector table
51  // - 50 cycles spent in PCINT vector to save context and call handler
52  // - 4 cycles spent in virtual handler thunk
53  // - 4N + 4 in delay
54  // - 18 cycles until first bit sample read
55  // For sampling of first bit we wait until middle of first bit
56  start_bit_rx_time_ = compute_delay(3 * bit_time / 2, 3 + 2 + 4 + 2 + 50 + 4 + 4 + 18);
57 
58  // 16+4N cycles elapse between processing of each bit
59  interbit_rx_time_ = compute_delay(bit_time, 16);
60 
61  if (parity != Parity::NONE)
62  {
63  // When parity must be checked, the number of cycles between last data bit sampled and parity bit sample is:
64  // - 20 cycles + 4n delay
65  parity_bit_rx_time_ = compute_delay(bit_time, 20);
66  // If no error (push) then the following cycles will get executed until we re-enable PCI
67  // - 21 cycles before delay loop
68  // - 41 cycles in push_
69  // - 4N cycles in delay loop
70  // - 5 additional cycles to make sure we passed the edge of stop bit
71  stop_bit_rx_time_push_ = compute_delay(bit_time / 2, 21 + 41 + 5);
72  // If an error occurred (no push) then the following cycles will get executed until we re-enable PCI
73  // - 26 cycles before delay loop
74  // - 4N cycles in delay loop
75  // - 5 additional cycles to make sure we passed the edge of stop bit
76  stop_bit_rx_time_no_push_ = compute_delay(bit_time / 2, 26 + 5);
77  }
78  else
79  {
80  // If no error (push) then the following cycles will get executed until we re-enable PCI
81  // - 24 cycles before delay loop
82  // - 41 cycles in push_
83  // - 4N cycles in delay loop
84  // - 5 additional cycles to make sure we passed the edge of stop bit
85  stop_bit_rx_time_push_ = compute_delay(bit_time / 2, 24 + 41 + 5);
86  }
87 
88  // We don't care about actual stop bit time, we just ensure we are ready for PCI before the end on stop bit
89  // Additionally, 49 cycles elapse until next PCI can be handled
90 }

92 serial::Parity serial::soft::AbstractUATX::calculate_parity(Parity parity, uint8_t value)
93 {
94  if (parity == Parity::NONE) return Parity::NONE;
95  bool odd = false;
96  while (value)
97  {
98  if (value & 0x01) odd = !odd;
99  value >>= 1;
100  }
101  return (odd ? Parity::ODD : Parity::EVEN);
102 }
