440  const char* justify(char* input, char filler = ' ')
441  {
442  uint8_t width = (width_ >= 0 ? width_ : -width_);
443  if (strlen(input) < width)
444  {
445  uint8_t add = width - strlen(input);
446  memmove(input + add, input, strlen(input) + 1);
447  memset(input, filler, add);
448  }
449  return input;
450  }
451  char filler()
452  {
453  switch (base_)
454  {
455  case Base::bin:
456  case Base::hex:
457  case Base::oct:
458  return '0';
459  case Base::dec:
460  return ' ';
461  }
462  }
