442  const char* justify(char* input, char filler = ' ')
443  {
444  uint8_t width = (_width >= 0 ? _width : -_width);
445  if (strlen(input) < width)
446  {
447  uint8_t add = width - strlen(input);
448  memmove(input + add, input, strlen(input) + 1);
449  memset(input, filler, add);
450  }
451  return input;
452  }
453  char filler()
454  {
455  switch (_base)
456  {
457  case Base::bin:
458  case Base::hex:
459  case Base::oct:
460  return '0';
461  case Base::dec:
462  return ' ';
463  }
464  }
