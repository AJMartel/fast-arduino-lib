506  template<uint8_t PCI_NUM_, board::InterruptPin PCIPIN1_, board::InterruptPin... PCIPINS_>
507  static void check_pci_pins()
508  {
509  static_assert(board_traits::PCI_trait<PCI_NUM_>::PORT != board::Port::NONE, "PORT must support PCI");
510  constexpr board::DigitalPin PIN = board::PCI_PIN<PCIPIN1_>();
511  static_assert(board_traits::DigitalPin_trait<PIN>::PORT == board_traits::PCI_trait<PCI_NUM_>::PORT,
512  "PIN port must match PCI_NUM port");
513  static_assert(
514  _BV(board_traits::DigitalPin_trait<PIN>::BIT) & board_traits::PCI_trait<PCI_NUM_>::PCI_MASK,
515  "PIN must be a PCINT pin");
516  // Check other pins
517  check_pci_pins<PCI_NUM_, PCIPINS_...>();
518  }
519 
520  template<uint8_t PCI_NUM_, typename HANDLER_, void (HANDLER_::*CALLBACK_)(), board::InterruptPin... PCIPINS_>
521  static void pci_method()
522  {
523  // Check pin is compliant
524  check_pci_pins<PCI_NUM_, PCIPINS_...>();
525  // Call handler back
526  interrupt::CallbackHandler<void (HANDLER_::*)(), CALLBACK_>::call();
527  }
528 
529  template<uint8_t PCI_NUM_, void (*CALLBACK_)(), board::InterruptPin... PCIPINS_> static void pci_function()
530  {
531  // Check pin is compliant
532  check_pci_pins<PCI_NUM_, PCIPINS_...>();
533  // Call handler back
534  CALLBACK_();
535  }
536  };
