486  struct isr_handler_pci
487  {
488  template<uint8_t PCI_NUM_> static void check_pci_pins() {}

490  template<uint8_t PCI_NUM_, board::DigitalPin PCIPIN1_, board::DigitalPin... PCIPINS_>
491  static void check_pci_pins()
492  {
493  static_assert(board_traits::PCI_trait<PCI_NUM_>::PORT != board::Port::NONE, "PORT must support PCI");
494  static_assert(board_traits::DigitalPin_trait<PCIPIN1_>::PORT == board_traits::PCI_trait<PCI_NUM_>::PORT,
495  "PIN port must match PCI_NUM port");
496  static_assert(
497  _BV(board_traits::DigitalPin_trait<PCIPIN1_>::BIT) & board_traits::PCI_trait<PCI_NUM_>::PCI_MASK,
498  "PIN must be a PCINT pin");
499  // Check other pins
500  check_pci_pins<PCI_NUM_, PCIPINS_...>();
501  }
502 
503  template<uint8_t PCI_NUM_, typename HANDLER_, void (HANDLER_::*CALLBACK_)(), board::DigitalPin... PCIPINS_>
504  static void pci_method()
505  {
506  // Check pin is compliant
507  check_pci_pins<PCI_NUM_, PCIPINS_...>();
508  // Call handler back
509  interrupt::CallbackHandler<void (HANDLER_::*)(), CALLBACK_>::call();
510  }
511 
512  template<uint8_t PCI_NUM_, void (*CALLBACK_)(), board::DigitalPin... PCIPINS_> static void pci_function()
513  {
514  // Check pin is compliant
515  check_pci_pins<PCI_NUM_, PCIPINS_...>();
516  // Call handler back
517  CALLBACK_();
518  }
519  };
521 }
