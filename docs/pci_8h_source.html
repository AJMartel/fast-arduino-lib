522  struct isr_handler_pci
523  {
524  template<uint8_t PCI_NUM_> static void check_pci_pins() {}

526  template<uint8_t PCI_NUM_, board::InterruptPin PCIPIN1_, board::InterruptPin... PCIPINS_>
527  static void check_pci_pins()
528  {
529  static_assert(board_traits::PCI_trait<PCI_NUM_>::PORT != board::Port::NONE, "PORT must support PCI");
530  constexpr board::DigitalPin PIN = board::PCI_PIN<PCIPIN1_>();
531  static_assert(board_traits::DigitalPin_trait<PIN>::PORT == board_traits::PCI_trait<PCI_NUM_>::PORT,
532  "PIN port must match PCI_NUM port");
533  static_assert(
534  _BV(board_traits::DigitalPin_trait<PIN>::BIT) & board_traits::PCI_trait<PCI_NUM_>::PCI_MASK,
535  "PIN must be a PCINT pin");
536  // Check other pins
537  check_pci_pins<PCI_NUM_, PCIPINS_...>();
538  }
539 
540  template<uint8_t PCI_NUM_, typename HANDLER_, void (HANDLER_::*CALLBACK_)(), board::InterruptPin... PCIPINS_>
541  static void pci_method()
542  {
543  // Check pin is compliant
544  check_pci_pins<PCI_NUM_, PCIPINS_...>();
545  // Call handler back
546  interrupt::CallbackHandler<void (HANDLER_::*)(), CALLBACK_>::call();
547  }
548 
549  template<uint8_t PCI_NUM_, void (*CALLBACK_)(), board::InterruptPin... PCIPINS_> static void pci_function()
550  {
551  // Check pin is compliant
552  check_pci_pins<PCI_NUM_, PCIPINS_...>();
553  // Call handler back
554  CALLBACK_();
555  }
556  };
558 }
