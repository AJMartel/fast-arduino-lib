60  bool set_datetime(tm& datetime)
61  {
62  // 1st convert datetime for DS1307 (BCD)
63  datetime.tm_sec = utils::binary_to_bcd(datetime.tm_sec);
64  datetime.tm_min = utils::binary_to_bcd(datetime.tm_min);
65  datetime.tm_hour = utils::binary_to_bcd(datetime.tm_hour);
66  datetime.tm_mday = utils::binary_to_bcd(datetime.tm_mday);
67  datetime.tm_mon = utils::binary_to_bcd(datetime.tm_mon);
68  datetime.tm_year = utils::binary_to_bcd(datetime.tm_year);
69  // send register address to write to (0)
70  // send datetime at address 0
71  return write(DEVICE_ADDRESS, TIME_ADDRESS, i2c::BusConditions::START_NO_STOP) == i2c::Status::OK
72  && write(DEVICE_ADDRESS, datetime, i2c::BusConditions::NO_START_STOP) == i2c::Status::OK;
73  }
74 
75  bool get_datetime(tm& datetime)
76  {
77  // send register address to read from (0)
78  // read datetime at address 0
79  if (write(DEVICE_ADDRESS, TIME_ADDRESS, i2c::BusConditions::START_NO_STOP) == i2c::Status::OK
80  && read(DEVICE_ADDRESS, datetime, i2c::BusConditions::REPEAT_START_STOP) == i2c::Status::OK)
81  {
82  // convert DS1307 output (BCD) to integer type
83  datetime.tm_sec = utils::bcd_to_binary(datetime.tm_sec);
84  datetime.tm_min = utils::bcd_to_binary(datetime.tm_min);
85  datetime.tm_hour = utils::bcd_to_binary(datetime.tm_hour);
86  datetime.tm_mday = utils::bcd_to_binary(datetime.tm_mday);
87  datetime.tm_mon = utils::bcd_to_binary(datetime.tm_mon);
88  datetime.tm_year = utils::bcd_to_binary(datetime.tm_year);
89  return true;
90  }
91  return false;
92  }

101  bool enable_output(SquareWaveFrequency frequency = SquareWaveFrequency::FREQ_1HZ)
102  {

114  && write(DEVICE_ADDRESS, control, i2c::BusConditions::NO_START_STOP) == i2c::Status::OK;

117  static constexpr uint8_t ram_size()
118  {
119  return RAM_SIZE;
120  }
121  bool set_ram(uint8_t address, uint8_t data)
122  {
123  address += RAM_START;
124  if (address < RAM_END)
125  return write(DEVICE_ADDRESS, address, i2c::BusConditions::START_NO_STOP) == i2c::Status::OK
126  && write(DEVICE_ADDRESS, data, i2c::BusConditions::NO_START_STOP) == i2c::Status::OK;
127  else
128  return false;
129  }
130  uint8_t get_ram(uint8_t address)
131  {
132  address += RAM_START;
133  uint8_t data = 0;
134  if (address < RAM_END)
135  {
136  write(DEVICE_ADDRESS, address, i2c::BusConditions::START_NO_STOP);
137  read(DEVICE_ADDRESS, data, i2c::BusConditions::REPEAT_START_STOP);
138  }
139  return data;
140  }
