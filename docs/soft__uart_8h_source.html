117  template<board::DigitalPin DPIN> void UATX<DPIN>::write_(uint8_t value)
118  {
119  // Pre-calculate all what we need: parity bit
120  Parity parity_bit = calculate_parity(parity_, value);
121 
122  // Write start bit
123  tx_.clear();
124  // Wait before starting actual value transmission
125  _delay_loop_2(start_bit_tx_time_);
126  for (uint8_t bit = 0; bit < 8; ++bit)
127  {
128  if (value & 0x01)
129  tx_.set();
130  else
131  {
132  // Additional NOP to ensure set/clear are executed exactly at the same time (cycle)
133  asm volatile("NOP");
134  tx_.clear();
135  }
136  value >>= 1;
137  _delay_loop_2(interbit_tx_time_);
138  }
139  // Add parity if needed
140  if (parity_bit != Parity::NONE)
141  {
142  if (parity_bit == parity_)
143  tx_.clear();
144  else
145  tx_.set();
146  _delay_loop_2(interbit_tx_time_);
147  }
148  // Add stop bit
149  tx_.set();
150  _delay_loop_2(stop_bit_tx_time_);
151  }

240  template<board::DigitalPin RX> void UARX<RX>::on_pin_change()
241  {
242  // Check RX is low (start bit)
243  if (rx_.value()) return;
244  uint8_t value = 0;
245  bool odd = false;
246  UARTErrors_ errors;
247  errors.has_errors = 0;
248  // Wait for start bit to finish
249  _delay_loop_2(start_bit_rx_time_);
250  // Read first 7 bits
251  for (uint8_t i = 0; i < 7; ++i)
252  {
253  if (rx_.value())
254  {
255  value |= 0x80;
256  odd = !odd;
257  }
258  value >>= 1;
259  _delay_loop_2(interbit_rx_time_);
260  }
261  // Read last bit
262  if (rx_.value())
263  {
264  value |= 0x80;
265  odd = !odd;
266  }
267 
268  if (parity_ != Parity::NONE)
269  {
270  // Wait for parity bit TODO NEED SPECIFIC DELAY HERE
271  _delay_loop_2(parity_bit_rx_time_);
272  bool parity_bit = (parity_ == Parity::ODD ? !odd : odd);
273  // Check parity bit
274  errors.all_errors.parity_error = (rx_.value() != parity_bit);
275  }
276 
277  // Push value if no error
278  if (!errors.has_errors)
279  {
280  errors.all_errors.queue_overflow = !in_().queue().push_(value);
281  // Wait for 1st stop bit
282  _delay_loop_2(stop_bit_rx_time_push_);
283  }
284  else
285  {
286  errors_ = errors;
287  // Wait for 1st stop bit
288  _delay_loop_2(stop_bit_rx_time_no_push_);
289  }
290  // Clear PCI interrupt to remove pending PCI occurred during this method and to detect next start bit
291  if (PIN_TRAIT::IS_INT)
292  int_->clear_();
293  else
294  pci_->clear_();
295  }
