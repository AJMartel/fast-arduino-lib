183  template<board::DigitalPin DPIN> void UATX<DPIN>::write_(uint8_t value)
184  {
185  // Pre-calculate all what we need: parity bit
186  Parity parity_bit = calculate_parity(parity_, value);
187 
188  // Write start bit
189  tx_.clear();
190  // Wait before starting actual value transmission
191  _delay_loop_2(start_bit_tx_time_);
192  for (uint8_t bit = 0; bit < 8; ++bit)
193  {
194  if (value & 0x01)
195  tx_.set();
196  else
197  {
198  // Additional NOP to ensure set/clear are executed exactly at the same time (cycle)
199  asm volatile("NOP");
200  tx_.clear();
201  }
202  value >>= 1;
203  _delay_loop_2(interbit_tx_time_);
204  }
205  // Add parity if needed
206  if (parity_bit != Parity::NONE)
207  {
208  if (parity_bit == parity_)
209  tx_.clear();
210  else
211  tx_.set();
212  _delay_loop_2(interbit_tx_time_);
213  }
214  // Add stop bit
215  tx_.set();
216  _delay_loop_2(stop_bit_tx_time_);
217  }

250  template<board::DigitalPin RX_> void AbstractUARX::pin_change()
251  {
252  using RX = gpio::FastPinType<RX_>;
253  // Check RX is low (start bit)
254  if (RX::value()) return;
255  uint8_t value = 0;
256  bool odd = false;
257  Errors errors;
258  errors.has_errors = 0;
259  // Wait for start bit to finish
260  _delay_loop_2(start_bit_rx_time_);
261  // Read first 7 bits
262  for (uint8_t i = 0; i < 7; ++i)
263  {
264  if (RX::value())
265  {
266  value |= 0x80;
267  odd = !odd;
268  }
269  value >>= 1;
270  _delay_loop_2(interbit_rx_time_);
271  }
272  // Read last bit
273  if (RX::value())
274  {
275  value |= 0x80;
276  odd = !odd;
277  }
278 
279  if (parity_ != Parity::NONE)
280  {
281  // Wait for parity bit TODO NEED SPECIFIC DELAY HERE
282  _delay_loop_2(parity_bit_rx_time_);
283  bool parity_bit = (parity_ == Parity::ODD ? !odd : odd);
284  // Check parity bit
285  errors.parity_error = (RX::value() != parity_bit);
286  }
287 
288  // Push value if no error
289  if (!errors.has_errors)
290  {
291  errors.queue_overflow = !in_().queue().push_(value);
292  // Wait for 1st stop bit
293  _delay_loop_2(stop_bit_rx_time_push_);
294  }
295  else
296  {
297  errors_ = errors;
298  // Wait for 1st stop bit
299  _delay_loop_2(stop_bit_rx_time_no_push_);
300  }
301  }
