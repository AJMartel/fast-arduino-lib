183  template<board::DigitalPin DPIN> void UATX<DPIN>::write_(uint8_t value)
184  {
185  // Pre-calculate all what we need: parity bit
186  Parity parity_bit = calculate_parity(parity_, value);
187 
188  // Write start bit
189  tx_.clear();
190  // Wait before starting actual value transmission
191  _delay_loop_2(start_bit_tx_time_);
192  for (uint8_t bit = 0; bit < 8; ++bit)
193  {
194  if (value & 0x01)
195  tx_.set();
196  else
197  {
198  // Additional NOP to ensure set/clear are executed exactly at the same time (cycle)
199  asm volatile("NOP");
200  tx_.clear();
201  }
202  value >>= 1;
203  _delay_loop_2(interbit_tx_time_);
204  }
205  // Add parity if needed
206  if (parity_bit != Parity::NONE)
207  {
208  if (parity_bit == parity_)
209  tx_.clear();
210  else
211  tx_.set();
212  _delay_loop_2(interbit_tx_time_);
213  }
214  // Add stop bit
215  tx_.set();
216  _delay_loop_2(stop_bit_tx_time_);
217  }

384  template<board::DigitalPin RX> void UARX<RX>::on_pin_change()
385  {
386  // Check RX is low (start bit)
387  if (rx_.value()) return;
388  uint8_t value = 0;
389  bool odd = false;
390  Errors errors;
391  errors.has_errors = 0;
392  // Wait for start bit to finish
393  _delay_loop_2(start_bit_rx_time_);
394  // Read first 7 bits
395  for (uint8_t i = 0; i < 7; ++i)
396  {
397  if (rx_.value())
398  {
399  value |= 0x80;
400  odd = !odd;
401  }
402  value >>= 1;
403  _delay_loop_2(interbit_rx_time_);
404  }
405  // Read last bit
406  if (rx_.value())
407  {
408  value |= 0x80;
409  odd = !odd;
410  }
411 
412  if (parity_ != Parity::NONE)
413  {
414  // Wait for parity bit TODO NEED SPECIFIC DELAY HERE
415  _delay_loop_2(parity_bit_rx_time_);
416  bool parity_bit = (parity_ == Parity::ODD ? !odd : odd);
417  // Check parity bit
418  errors.parity_error = (rx_.value() != parity_bit);
419  }
420 
421  // Push value if no error
422  if (!errors.has_errors)
423  {
424  errors.queue_overflow = !in_().queue().push_(value);
425  // Wait for 1st stop bit
426  _delay_loop_2(stop_bit_rx_time_push_);
427  }
428  else
429  {
430  errors_ = errors;
431  // Wait for 1st stop bit
432  _delay_loop_2(stop_bit_rx_time_no_push_);
433  }
434  // Clear PCI interrupt to remove pending PCI occurred during this method and to detect next start bit
435  if (PIN_TRAIT::IS_INT)
436  int_->clear_();
437  else
438  pci_->clear_();
439  }
