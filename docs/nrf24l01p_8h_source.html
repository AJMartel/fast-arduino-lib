588  template<board::DigitalPin CSN, board::DigitalPin CE> void NRF24L01<CSN, CE>::begin()
589  {
590  // Setup hardware features, channel, bitrate, retransmission, dynamic payload
591  write_register(Register::FEATURE, (_BV(EN_DPL) | _BV(EN_ACK_PAY) | _BV(EN_DYN_ACK)));
592  write_register(Register::RF_CH, channel_);
593  write_register(Register::RF_SETUP, RF_DR_2MBPS | RF_PWR_0DBM);
594  write_register(Register::SETUP_RETR, (DEFAULT_ARD << ARD) | (DEFAULT_ARC << ARC));
595  write_register(Register::DYNPD, DPL_PA);
596 
597  // Setup hardware receive pipes address; network (16-bit), device (8-bit)
598  // P0: auto-acknowledge (see set_transmit_mode)
599  // P1: node address<network:device> with auto-acknowledge
600  // P2: broadcast<network:0>
601  addr_t rx_addr = addr_;
602  write_register(Register::SETUP_AW, AW_3BYTES);
603  write_register(Register::RX_ADDR_P1, &rx_addr, sizeof(rx_addr));
604  write_register(Register::RX_ADDR_P2, BROADCAST);
605  write_register(Register::EN_RXADDR, _BV(ERX_P2) | _BV(ERX_P1));
606  write_register(Register::EN_AA, _BV(ENAA_P1) | _BV(ENAA_P0));
607 
608  // Ready to go
609  powerup();
610  }

612  template<board::DigitalPin CSN, board::DigitalPin CE> void NRF24L01<CSN, CE>::powerup()
613  {
614  if (state_ != State::POWER_DOWN_STATE) return;
615  ce_.clear();
616 
617  // Setup configuration for powerup and clear interrupts
618  write_register(Register::CONFIG, _BV(EN_CRC) | _BV(CRCO) | _BV(PWR_UP));
619  time::delay_ms(Tpd2stby_ms);
620  state_ = State::STANDBY_STATE;
621 
622  // Flush status
623  write_register(Register::STATUS, _BV(RX_DR) | _BV(TX_DS) | _BV(MAX_RT));
624  write_command(Command::FLUSH_TX);
625  write_command(Command::FLUSH_RX);
626  }

643  template<board::DigitalPin CSN, board::DigitalPin CE>
644  int NRF24L01<CSN, CE>::send(uint8_t dest, uint8_t port, const void* buf, size_t len)
645  {
646  if (buf == 0 && len > 0) return errors::EINVAL;
647  if (len > PAYLOAD_MAX) return errors::EMSGSIZE;
648 
649  // Setting transmit destination first (needs to ensure standby mode)
650  transmit_mode(dest);
651 
652  // Write source address and payload to the transmit fifo
653  Command command = ((dest != BROADCAST) ? Command::W_TX_PAYLOAD : Command::W_TX_PAYLOAD_NO_ACK);
654  this->start_transfer();
655  status_ = this->transfer(uint8_t(command));
656  this->transfer(addr_.device);
657  this->transfer(port);
658  this->transfer((uint8_t*) buf, len);
659  this->end_transfer();
660 
661  trans_ += 1;
662 
663  // Check for auto-acknowledge pipe(0), and address setup and enable
664  if (dest != BROADCAST)
665  {
666  addr_t tx_addr(addr_.network, dest);
667  write_register(Register::RX_ADDR_P0, &tx_addr, sizeof(tx_addr));
668  write_register(Register::EN_RXADDR, _BV(ERX_P2) | _BV(ERX_P1) | _BV(ERX_P0));
669  }
670 
671  // Wait for transmission
672  status_t status = 0;
673  while (true)
674  {
675  status = read_status();
676  if (status.tx_ds || status.max_rt) break;
677  time::yield();
678  }
679 
680  bool data_sent = status.tx_ds;
681 
682  // Check for auto-acknowledge pipe(0) disable
683  if (dest != BROADCAST) write_register(Register::EN_RXADDR, _BV(ERX_P2) | _BV(ERX_P1));
684 
685  // Reset status bits
686  write_register(Register::STATUS, _BV(TX_DS) | _BV(MAX_RT));
687 
688  // Read retransmission counter and update
689  observe_tx_t observe = read_observe_tx();
690  retrans_ += observe.arc_cnt;
691 
692  // Check that the message was delivered
693  if (data_sent) return len;
694 
695  // Failed to deliver
696  write_command(Command::FLUSH_TX);
697  drops_ += 1;
698 
699  return errors::EIO;
700  }

702  template<board::DigitalPin CSN, board::DigitalPin CE>
703  int NRF24L01<CSN, CE>::recv(uint8_t& src, uint8_t& port, void* buf, size_t size, uint32_t ms)
704  {
705  // Run in receive mode
706  receive_mode();
707 
708  // Check if there is data available on any pipe
709  uint32_t start = time::millis();
710  while (!available())
711  {
712  if ((ms != 0) && (time::since(start) > ms)) return errors::ETIME;
713  time::yield();
714  }
715 
716  // Try and read payload from FIFO
717  return read_fifo_payload(src, port, buf, size);
718  }

732  template<board::DigitalPin CSN, board::DigitalPin CE> void NRF24L01<CSN, CE>::transmit_mode(uint8_t dest)
733  {
734  // Setup primary transmit address
735  addr_t tx_addr(addr_.network, dest);
736  write_register(Register::TX_ADDR, &tx_addr, sizeof(tx_addr));
737 
738  // Trigger the transmitter mode
739  if (state_ != State::TX_STATE)
740  {
741  ce_.clear();
742  write_register(Register::CONFIG, _BV(EN_CRC) | _BV(CRCO) | _BV(PWR_UP));
743  ce_.set();
744  }
745 
746  // Wait for the transmitter to become active
747  if (state_ == State::STANDBY_STATE) time::delay_us(Tstby2a_us);
748  state_ = State::TX_STATE;
749  }

751  template<board::DigitalPin CSN, board::DigitalPin CE> void NRF24L01<CSN, CE>::receive_mode()
752  {
753  // Check already in receive mode
754  if (state_ == State::RX_STATE) return;
755 
756  // Configure primary receiver mode
757  write_register(Register::CONFIG, _BV(EN_CRC) | _BV(CRCO) | _BV(PWR_UP) | _BV(PRIM_RX));
758  ce_.set();
759  if (state_ == State::STANDBY_STATE) time::delay_us(Tstby2a_us);
760  state_ = State::RX_STATE;
761  }

763  template<board::DigitalPin CSN, board::DigitalPin CE> bool NRF24L01<CSN, CE>::available()
764  {
765  // Check the receiver fifo
766  if (read_fifo_status().rx_empty) return false;
767 
768  // Sanity check the size of the payload. Might require a flush
769  if (read_command(Command::R_RX_PL_WID) <= DEVICE_PAYLOAD_MAX) return true;
770  write_command(Command::FLUSH_RX);
771  return false;
772  }

774  template<board::DigitalPin CSN, board::DigitalPin CE>
775  int NRF24L01<CSN, CE>::read_fifo_payload(uint8_t& src, uint8_t& port, void* buf, size_t size)
776  {
777  // Check for payload error from device (Tab. 20, pp. 51, R_RX_PL_WID)
778  uint8_t count = read_command(Command::R_RX_PL_WID) - 2;
779  if ((count > PAYLOAD_MAX) || (count > size))
780  {
781  write_command(Command::FLUSH_RX);
782  return errors::EMSGSIZE;
783  }
784 
785  // Data is available, check if this a broadcast or not
786  dest_ = (read_status().rx_p_no == 1 ? addr_.device : BROADCAST);
787 
788  // Read the source address, port and payload
789  this->start_transfer();
790  status_ = this->transfer(uint8_t(Command::R_RX_PAYLOAD));
791  src = this->transfer(0);
792  port = this->transfer(0);
793  this->transfer((uint8_t*) buf, count, uint8_t(Command::NOP));
794  this->end_transfer();
795  return count;
796  }
