318  template<board::DigitalPin CS> bool WinBond<CS>::wait_until_ready(uint16_t timeout_ms)
319  {
320  bool ready = false;
321  this->start_transfer();
322  this->transfer(0x05);
323  uint32_t start = time::millis();
324  while (true)
325  {
326  uint8_t status = this->transfer(0x00);
327  if (!(status & 0x01))
328  {
329  ready = true;
330  break;
331  }
332  if ((timeout_ms != 0) && (time::since(start) > timeout_ms)) break;
333  time::yield();
334  }
335  this->end_transfer();
336  return ready;
337  }

346  template<board::DigitalPin CS> uint64_t WinBond<CS>::read_unique_ID()
347  {
348  // Since the Read ID instruction must be followed by 4 dummy bytes before
349  // returning the 8 bytes ID, we must use a 9-bytes buffer and skip its
350  // first byte (the 3 other dummy bytes are already sent by send() as the
351  // 0 address)
352  struct PAYLOAD
353  {
354  uint8_t dummy;
355  uint64_t id;
356  };
357  PAYLOAD buffer;
358  send(0x4B, 0, (uint8_t*) &buffer, sizeof buffer);
359  // WinBond ID is big-endian (high byte first) but AVR is little-endian
360  // hence we need to convert result (using GCC builtin utility)
361  return __builtin_bswap64(buffer.id);
362  }
