230  inline RTTTime operator-(const RTTTime& a, const RTTTime& b)
231  {
232  if (a <= b) return RTTTime{0, 0};
233  uint32_t millis = a.millis - b.millis;
234  if (a.micros >= b.micros) return RTTTime{millis, a.micros - b.micros};
