109  template<typename T, typename TREF> bool Queue<T, TREF>::peek_(T& item) const
110  {
111  if (tail_ == head_) return false;
112  item = buffer_[head_];
113  return true;
114  }

116  template<typename T, typename TREF> uint8_t Queue<T, TREF>::peek_(T* buffer, uint8_t size) const
117  {
118  uint8_t actual = (tail_ - head_) & mask_;
119  if (size > actual) size = actual;
120  //TODO optimize copy (index calculation is simple if split in 2 parts)
121  for (uint8_t i = 0; i < size; ++i) buffer[i] = buffer_[(head_ + i) & mask_];
122  return size;
123  }

130  template<typename T, typename TREF> bool Queue<T, TREF>::push_(TREF item)
131  {
132  if ((head_ - tail_ - 1) & mask_)
133  {
134  buffer_[tail_] = item;
135  tail_ = (tail_ + 1) & mask_;
136  return true;
137  }
138  return false;
139  }

141  template<typename T, typename TREF> bool Queue<T, TREF>::pull_(T& item)
142  {
143  if (tail_ != head_)
144  {
145  item = buffer_[head_];
146  head_ = (head_ + 1) & mask_;
147  return true;
148  }
149  return false;
150  }
