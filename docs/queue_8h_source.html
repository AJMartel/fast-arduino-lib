458  template<typename T, typename TREF> uint8_t Queue<T, TREF>::peek_(T* buffer, uint8_t size) const
459  {
460  size = min(size, items());
461  if (size)
462  {
463  // Split peek in 2 parts if needed
464  if (head_ < tail_)
465  {
466  const T* source = &buffer_[head_];
467  for (uint8_t i = 0; i < size; ++i) *buffer++ = *source++;
468  }
469  else
470  {
471  uint8_t part_size = size_ - head_;
472  const T* source = &buffer_[head_];
473  for (uint8_t i = 0; i < part_size; ++i) *buffer++ = *source++;
474  part_size = size - part_size;
475  source = buffer_;
476  for (uint8_t i = 0; i < part_size; ++i) *buffer++ = *source++;
477  }
478  }
479  return size;
480  }
