168  bool begin(OperatingMode mode = OperatingMode::SINGLE, Gain gain = Gain::GAIN_1_3GA,
169  DataOutput rate = DataOutput::RATE_15HZ, SamplesAveraged samples = SamplesAveraged::ONE_SAMPLE,
170  MeasurementMode measurement = MeasurementMode::NORMAL)
171  {
172  gain_ = GAIN(gain);
173  return write_register(CONFIG_REG_A, uint8_t(measurement) | uint8_t(rate) | uint8_t(samples))
174  && write_register(CONFIG_REG_B, uint8_t(gain)) && write_register(MODE_REG, uint8_t(mode));
175  }

214  bool magnetic_fields(MagneticFields& fields)
215  {
216  if (this->write(DEVICE_ADDRESS, OUTPUT_REG_1, i2c::BusConditions::START_NO_STOP) == i2c::Status::OK
217  && this->read(DEVICE_ADDRESS, fields, i2c::BusConditions::REPEAT_START_STOP) == i2c::Status::OK)
218  {
219  utils::swap_bytes(fields.x);
220  utils::swap_bytes(fields.y);
221  utils::swap_bytes(fields.z);
222  return true;
223  }
224  else
225  return false;
226  }

257  bool write_register(uint8_t address, uint8_t value)
258  {
259  return (this->write(DEVICE_ADDRESS, address, i2c::BusConditions::START_NO_STOP) == i2c::Status::OK
260  && this->write(DEVICE_ADDRESS, value, i2c::BusConditions::NO_START_STOP) == i2c::Status::OK);

266  && this->read(DEVICE_ADDRESS, value, i2c::BusConditions::REPEAT_START_STOP) == i2c::Status::OK);

274  static constexpr uint16_t GAIN(Gain gain)
275  {

281  gain == Gain::GAIN_4_7GA ? 390 :
282  gain == Gain::GAIN_5_6GA ? 330 : 230);
283  }
