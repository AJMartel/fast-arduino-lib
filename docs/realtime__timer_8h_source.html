444  template<uint8_t TIMER_NUM_, typename HANDLER_, void (HANDLER_::*CALLBACK_)(uint32_t)> static void rtt_method()
445  {
446  static constexpr board::Timer NTIMER = isr_handler::check_timer<TIMER_NUM_>();
447  auto handler = interrupt::HandlerHolder<RTT<NTIMER>>::handler();
448  handler->on_timer();
449  interrupt::CallbackHandler<void (HANDLER_::*)(uint32_t), CALLBACK_>::call(handler->millis());
450  }

454  static constexpr board::Timer NTIMER = isr_handler::check_timer<TIMER_NUM_>();
455  auto handler = interrupt::HandlerHolder<RTT<NTIMER>>::handler();
456  handler->on_timer();
457  CALLBACK_(handler->millis());
