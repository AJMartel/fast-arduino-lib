263  template<typename T> bool write_register(uint8_t address, T value)
264  {
265  using namespace i2c::Status;
266  return this->write(device_, address, BusCond::START_NO_STOP) == OK
267  && this->write(device_, value, BusCond::NO_START_STOP) == OK;
268  }
269 
270  template<typename T> bool read_register(uint8_t address, T& value)
271  {
272  using namespace i2c::Status;
273  return this->write(device_, address, BusCond::START_NO_STOP) == OK
274  && this->read(device_, value, BusCond::REPEAT_START_STOP) == OK;
275  }
