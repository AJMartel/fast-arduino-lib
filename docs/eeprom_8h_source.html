435  // In order to optimize write time we read current byte first, then compare it with new value
436  // Then we choose between erase, write and erase+write based on comparison
437  // This approach is detailed in ATmel note AVR103: Using the EEPROM Programming Modes
438  // http://www.atmel.com/images/doc2578.pdf
439  inline static void write_byte(uint16_t address, uint8_t value)
440  {
441  EEAR = address;
442  EECR = _BV(EERE);
443  uint8_t old_value = EEDR;
444  uint8_t diff = old_value ^ value;
445  if (diff & value)
446  {
447  // Some bits need to be erased (ie set to 1)
448  if (value == 0xFF)
449  // Erase only
450  EECR = EEPM0;
451  else
452  // Combine Erase/Write operation
453  EECR = 0;
454  }
455  else
456  {
457  // No bit to be erased
458  if (diff)
459  // Some bits to be programmed (set to 0): Write only
460  EECR = EEPM1;
461  else
462  // old value == new value => do nothing
463  return;
464  }
465  EEDR = value;
466  synchronized
467  {
468  EECR |= _BV(EEMPE);
469  EECR |= _BV(EEPE);
470  }
471  }
472 
473  inline static bool erase_address(uint16_t address)
474  {
475  EEAR = address;
476  EECR = _BV(EERE);
477  uint8_t value = EEDR;
478  // Some bits need to be erased (ie set to 1)
479  if (value == 0xFF) return false;
480  EECR = EEPM0;
481  EEDR = 0xFF;
482  synchronized
483  {
484  EECR |= _BV(EEMPE);
485  EECR |= _BV(EEPE);
486  }
487  return true;
488  }

661  void erase()
662  {
663  // First remove all pending writes
664  synchronized
665  {
666  buffer_.clear_();
667  current_.size = 0;
668  }
669  // Wait until current byte write is finished
670  wait_until_done();
671  synchronized
672  {
673  // Start erase
674  erase_ = true;
675  done_ = false;
676  current_.address = 0;
677  current_.size = size();
678  // Start transmission if not done yet
679  EECR = _BV(EERIE);
680  }
681  }

704  //TODO should better be private
706  bool on_ready()
707  {
708  if (erase_)
709  {
710  if (current_.size)
711  erase_next();
712  else
713  {
714  // All erases are finished
715  erase_ = false;
716  // Mark all EEPROM work as finished if no write is pending in the queue
717  if (buffer_.empty_())
718  {
719  done_ = true;
720  EECR = 0;
721  }
722  }
723  }
724  else if (current_.size)
725  // There is one item being currently written, write next byte
726  write_next();
727  else if (!buffer_.empty_())
728  {
729  // Current item is finished writing but there is another item to be written in the queue
730  // Get new item and start transmission of first byte
731  current_ = next_item();
732  write_next();
733  }
734  else
735  {
736  // All writes are finished
737  done_ = true;
738  EECR = 0;
739  }
740  return done_;
741  }

747  void write_next()
748  {
749  uint8_t value;
750  buffer_.pull_(value);
751  EEPROM::write_byte(current_.address++, value);
752  --current_.size;
753  EECR |= _BV(EERIE);
754  }

763  bool write_data(uint16_t address, uint8_t* value, uint16_t size)
764  {
765  // First check if there is enough space in buffer_ for this queued write
766  if ((buffer_.free_() < size + ITEM_SIZE) || !size) return false;
767  done_ = false;
768  // Add new queued write to buffer
769  buffer_.push_(WriteItem::value1(address, size));
770  buffer_.push_(WriteItem::value2(address, size));
771  buffer_.push_(WriteItem::value3(address, size));
772  for (uint16_t i = 0; i < size; ++i) buffer_.push_(*value++);
773 
774  // Start transmission if not done yet
775  EECR = _BV(EERIE);
776  return true;
777  }
