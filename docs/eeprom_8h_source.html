434  // In order to optimize write time we read current byte first, then compare it with new value
435  // Then we choose between erase, write and erase+write based on comparison
436  // This approach is detailed in ATmel note AVR103: Using the EEPROM Programming Modes
437  // http://www.atmel.com/images/doc2578.pdf
438  inline static void write_byte(uint16_t address, uint8_t value)
439  {
440  EEAR_ = address;
441  EECR_ = _BV(EERE);
442  uint8_t old_value = EEDR_;
443  uint8_t diff = old_value ^ value;
444  if (diff & value)
445  {
446  // Some bits need to be erased (ie set to 1)
447  if (value == 0xFF)
448  // Erase only
449  EECR_ = EEPM0;
450  else
451  // Combine Erase/Write operation
452  EECR_ = 0;
453  }
454  else
455  {
456  // No bit to be erased
457  if (diff)
458  // Some bits to be programmed (set to 0): Write only
459  EECR_ = EEPM1;
460  else
461  // old value == new value => do nothing
462  return;
463  }
464  EEDR_ = value;
465  synchronized
466  {
467  EECR_ |= _BV(EEMPE);
468  EECR_ |= _BV(EEPE);
469  }
470  }
471 
472  inline static bool erase_address(uint16_t address)
473  {
474  EEAR_ = address;
475  EECR_ = _BV(EERE);
476  uint8_t value = EEDR;
477  // Some bits need to be erased (ie set to 1)
478  if (value == 0xFF) return false;
479  EECR_ = EEPM0;
480  EEDR_ = 0xFF;
481  synchronized
482  {
483  EECR_ |= _BV(EEMPE);
484  EECR_ |= _BV(EEPE);
485  }
486  return true;
487  }

493  static constexpr const REG16 EEAR_{EEAR};
494  static constexpr const REG8 EECR_{EECR};
495  static constexpr const REG8 EEDR_{EEDR};
496  };

667  void erase()
668  {
669  // First remove all pending writes
670  synchronized
671  {
672  buffer_.clear_();
673  current_.size = 0;
674  }
675  // Wait until current byte write is finished
676  wait_until_done();
677  synchronized
678  {
679  // Start erase
680  erase_ = true;
681  done_ = false;
682  current_.address = 0;
683  current_.size = size();
684  // Start transmission if not done yet
685  EECR_ = _BV(EERIE);
686  }
687  }
688

697  {
698  while (!done_)
699  ;
700  }

708  }

711  static const uint16_t ITEM_SIZE = 3;

713  bool on_ready()
714  {
715  if (erase_)
716  {
717  if (current_.size)
718  erase_next();
719  else
720  {
721  // All erases are finished
722  erase_ = false;
723  // Mark all EEPROM work as finished if no write is pending in the queue
724  if (buffer_.empty_())
725  {
726  done_ = true;
727  EECR_ = 0;
728  }
729  }
730  }
731  else if (current_.size)
732  // There is one item being currently written, write next byte
733  write_next();
734  else if (!buffer_.empty_())
735  {
736  // Current item is finished writing but there is another item to be written in the queue
737  // Get new item and start transmission of first byte
738  current_ = next_item();
739  write_next();
740  }
741  else
742  {
743  // All writes are finished
744  done_ = true;
745  EECR_ = 0;
746  }
747  return done_;
748  }
749 
750  void write_next()
751  {
752  uint8_t value;
753  buffer_.pull_(value);
754  EEPROM::write_byte(current_.address++, value);
755  --current_.size;
756  EECR_ |= _BV(EERIE);
757  }
758 
759  void erase_next()
760  {
761  EEPROM::erase_address(current_.address++);
762  --current_.size;
763  EECR_ |= _BV(EERIE);
764  }

766  bool write_data(uint16_t address, uint8_t* value, uint16_t size)
767  {
768  // First check if there is enough space in buffer_ for this queued write
769  if ((buffer_.free_() < size + ITEM_SIZE) || !size) return false;
770  done_ = false;
771  // Add new queued write to buffer
772  buffer_.push_(WriteItem::value1(address, size));
773  buffer_.push_(WriteItem::value2(address, size));
774  buffer_.push_(WriteItem::value3(address, size));
775  for (uint16_t i = 0; i < size; ++i) buffer_.push_(*value++);
776 
777  // Start transmission if not done yet
778  EECR_ = _BV(EERIE);
779  return true;
780  }
