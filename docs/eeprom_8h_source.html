186  // In order to optimize write time we read current byte first, then compare it with new value
187  // Then we choose between erase, write and erase+write based on comparison
188  // This approach is detailed in ATmel note AVR103: Using the EEPROM Programming Modes
189  // http://www.atmel.com/images/doc2578.pdf
190  inline static void _write(uint16_t address, uint8_t value)
191  {
192  EEAR = address;
193  EECR = _BV(EERE);
194  uint8_t old_value = EEDR;
195  uint8_t diff = old_value ^ value;
196  if (diff & value)
197  {
198  // Some bits need to be erased (ie set to 1)
199  if (value == 0xFF)
200  // Erase only
201  EECR = EEPM0;
202  else
203  // Combine Erase/Write operation
204  EECR = 0;
205  }
206  else
207  {
208  // No bit to be erased
209  if (diff)
210  // Some bits to be programmed (set to 0): Write only
211  EECR = EEPM1;
212  else
213  // old value == new value => do nothing
214  return;
215  }
216  EEDR = value;
217  synchronized
218  {
219  EECR |= _BV(EEMPE);
220  EECR |= _BV(EEPE);
221  }
222  }

224  inline static bool _erase(uint16_t address)
225  {
226  EEAR = address;
227  EECR = _BV(EERE);
228  uint8_t value = EEDR;
229  // Some bits need to be erased (ie set to 1)
230  if (value == 0xFF) return false;
231  EECR = EEPM0;
232  EEDR = 0xFF;
233  synchronized
234  {
235  EECR |= _BV(EEMPE);
236  EECR |= _BV(EEPE);
237  }
238  return true;
239  }

288  void erase()
289  {
290  // First remove all pending writes
291  synchronized
292  {
293  _buffer._clear();
294  _current.size = 0;
295  }
296  // Wait until current byte write is finished
297  wait_until_done();
298  synchronized
299  {
300  // Start erase
301  _erase = true;
302  _done = false;
303  _current.address = 0;
304  _current.size = size();
305  // Start transmission if not done yet
306  EECR = _BV(EERIE);
307  }
308  }

320  void on_ready()
321  {
322  if (_erase)
323  {
324  if (_current.size)
325  erase_next();
326  else
327  {
328  // All erases are finished
329  _erase = false;
330  // Mark all EEPROM work as finished if no write is pending in the queue
331  if (_buffer._empty())
332  {
333  _done = true;
334  EECR = 0;
335  }
336  }
337  }
338  else if (_current.size)
339  // There is one item being currently written, write next byte
340  write_next();
341  else if (!_buffer._empty())
342  {
343  // Current item is finished writing but there is another item to be written in the queue
344  // Get new item and start transmission of first byte
345  _current = next_item();
346  write_next();
347  }
348  else
349  {
350  // All writes are finished
351  _done = true;
352  EECR = 0;
353  }
354  }

375  bool _write(uint16_t address, uint8_t* value, uint16_t size)
376  {
377  // First check if there is enough space in _buffer for this queued write
378  if ((_buffer._free() < size + ITEM_SIZE) || !size)
379  return false;
380  _done = false;
381  // Add new queued write to buffer
382  _buffer._push(WriteItem::value1(address, size));
383  _buffer._push(WriteItem::value2(address, size));
384  _buffer._push(WriteItem::value3(address, size));
385  for (uint16_t i = 0; i < size; ++i)
386  _buffer._push(*value++);
387 
388  // Start transmission if not done yet
389  EECR = _BV(EERIE);
390  return true;
391  }
