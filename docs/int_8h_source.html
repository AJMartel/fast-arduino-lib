263  template<uint8_t INT_NUM_, board::DigitalPin INT_PIN_, typename HANDLER_, void (HANDLER_::*CALLBACK_)()>
264  static void int_method()
265  {
266  // Check pin is compliant
267  check_int_pin<INT_NUM_, INT_PIN_>();
268  // Call handler back
269  interrupt::CallbackHandler<void (HANDLER_::*)(), CALLBACK_>::call();
270  }

272  template<uint8_t INT_NUM_, board::DigitalPin INT_PIN_, void (*CALLBACK_)()> static void int_function()
273  {
274  // Check pin is compliant
275  check_int_pin<INT_NUM_, INT_PIN_>();
