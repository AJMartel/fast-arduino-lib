82  SAMPLE_TYPE sample()
83  {
84  // First ensure that any pending sampling is finished
85  GLOBAL_TRAIT::ADCSRA_.loop_until_bit_clear(ADSC);
86  // Setup multiplexer selection and start conversion
87  GLOBAL_TRAIT::ADMUX_ = VREF_TRAIT::MASK | TYPE_TRAIT::ADLAR1 | TRAIT::MUX_MASK1;
88  GLOBAL_TRAIT::ADCSRB_ = TRAIT::MUX_MASK2 | TYPE_TRAIT::ADLAR2;
89 
90  // The following delay is necessary for bandgap ADC, strangely 70us should be enough (datasheet)
91  // but this works only when no other ADC is used "at the same time"
92  // In this situation, a delay of minimum 400us seems necessary to ensure bandgap reference voltage is stabilized
93  if (TRAIT::IS_BANDGAP)
94  time::delay_us(BG_STABILIZATION_DELAY_US);
95 
96  GLOBAL_TRAIT::ADCSRA_ = _BV(ADEN) | _BV(ADSC) | TRAIT::MUX_MASK2 | FREQ_TRAIT::PRESCALER_MASK;
97  // Wait until sampling is done
98  GLOBAL_TRAIT::ADCSRA_.loop_until_bit_clear(ADSC);
99  // Should we synchronize ADC reading?
100  return TYPE_TRAIT::_ADC;
101  }
