88  static constexpr SpeedSetup compute_speed(uint32_t rate)
89  {
90  const uint16_t double_rate = UBRR_double(rate);
91  if (double_rate < DOUBLE_SPEED_RATE_LIMIT)
92  return SpeedSetup(double_rate, true);
93  else
94  return SpeedSetup(UBRR_single(rate), false);
95  }
96 
97  private:
98  static constexpr const uint16_t DOUBLE_SPEED_RATE_LIMIT = 4096;
99 
100  static constexpr uint16_t UBRR_double(uint32_t rate)
101  {
102  return (F_CPU / 4 / rate - 1) / 2;
103  }
104  static constexpr uint16_t UBRR_single(uint32_t rate)
105  {
106  return (F_CPU / 8 / rate - 1) / 2;
107  }

162  inline void begin(uint32_t rate, Parity parity = Parity::NONE, StopBits stop_bits = StopBits::ONE)
163  {
164  AbstractUART::SpeedSetup setup = AbstractUART::compute_speed(rate);
165  synchronized
166  {
167  TRAIT::UBRR = setup.ubrr_value;
168  TRAIT::UCSRA = (setup.u2x ? TRAIT::U2X_MASK : 0);
169  TRAIT::UCSRB = TRAIT::TX_ENABLE_MASK | TRAIT::UDRIE_MASK;
170  TRAIT::UCSRC = TRAIT::UCSRC_value(parity, stop_bits);
171  }
172  }

193  private:
194  inline void data_register_empty()
195  {
196  errors_.has_errors = 0;
197  char value;
198  if (queue().pull_(value))
199  TRAIT::UDR = value;
200  else
201  {
202  transmitting_ = false;
203  // Clear UDRIE to prevent UDR interrupt to go on forever
204  TRAIT::UCSRB &= ~TRAIT::UDRIE_MASK;
205  }
206  }
207 
208  protected:
210  // Listeners of events on the buffer
211  virtual void on_put() override
212  {
213  errors_.queue_overflow = ostreambuf::overflow();
214  synchronized
215  {
216  // Check if TX is not currently active, if so, activate it
217  if (!transmitting_)
218  {
219  // Yes, trigger TX
220  char value;
221  if (queue().pull_(value))
222  {
223  // Set UDR interrupt to be notified when we can send the next character
224  TRAIT::UCSRB |= TRAIT::UDRIE_MASK;
225  TRAIT::UDR = value;
226  transmitting_ = true;
227  }
228  }
229  }
230  }
232 
233  private:
234  bool transmitting_;
235  friend class UART<USART>;
236  friend struct isr_handler;
237  };
