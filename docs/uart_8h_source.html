196  inline void data_register_empty()
197  {
198  _errors.has_errors = 0;
199  char value;
200  if (out()._pull(value))
201  TRAIT::UDR = value;
202  else
203  {
204  _errors.all_errors.queue_overflow = true;
205  _transmitting = false;
206  // Clear UDRIE to prevent UDR interrupt to go on forever
207  TRAIT::UCSRB &= ~TRAIT::UDRIE_MASK;
208  }
209  }
211 
212  protected:
214  // Listeners of events on the buffer
215  virtual void on_put() override
216  {
217  synchronized
218  {
219  // Check if TX is not currently active, if so, activate it
220  if (!_transmitting)
221  {
222  // Yes, trigger TX
223  char value;
224  if (OutputBuffer::_pull(value))
225  {
226  // Set UDR interrupt to be notified when we can send the next character
227  TRAIT::UCSRB |= TRAIT::UDRIE_MASK;
228  TRAIT::UDR = value;
229  _transmitting = true;
230  }
231  }
232  }
233  }
234  virtual void on_overflow(UNUSED char c) override
235  {
236  _errors.all_errors.queue_overflow = true;
