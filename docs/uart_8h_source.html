202  inline void data_register_empty()
203  {
204  errors_.has_errors = 0;
205  char value;
206  if (out().pull_(value))
207  TRAIT::UDR = value;
208  else
209  {
210  errors_.all_errors.queue_overflow = true;
211  transmitting_ = false;
212  // Clear UDRIE to prevent UDR interrupt to go on forever
213  TRAIT::UCSRB &= ~TRAIT::UDRIE_MASK;
214  }
215  }
217 
218  protected:
220  // Listeners of events on the buffer
221  virtual void on_put() override
222  {
223  synchronized
224  {
225  // Check if TX is not currently active, if so, activate it
226  if (!transmitting_)
227  {
228  // Yes, trigger TX
229  char value;
230  if (OutputBuffer::pull_(value))
231  {
232  // Set UDR interrupt to be notified when we can send the next character
233  TRAIT::UCSRB |= TRAIT::UDRIE_MASK;
234  TRAIT::UDR = value;
235  transmitting_ = true;
236  }
237  }
238  }
239  }
240  virtual void on_overflow(UNUSED char c) override
241  {
242  errors_.all_errors.queue_overflow = true;
243  }
245 
246  private:
247  bool transmitting_;
248  };

342  inline void data_receive_complete()
343  {
344  char status = TRAIT::UCSRA;
345  errors_.all_errors.data_overrun = status & TRAIT::DOR_MASK;
346  errors_.all_errors.frame_error = status & TRAIT::FE_MASK;
347  errors_.all_errors.parity_error = status & TRAIT::UPE_MASK;
348  char value = TRAIT::UDR;
349  errors_.all_errors.queue_overflow = !in().push_(value);
