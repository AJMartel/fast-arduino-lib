193  inline void data_register_empty()
194  {
195  errors_.has_errors = 0;
196  char value;
197  if (queue().pull_(value))
198  TRAIT::UDR = value;
199  else
200  {
201  transmitting_ = false;
202  // Clear UDRIE to prevent UDR interrupt to go on forever
203  TRAIT::UCSRB &= ~TRAIT::UDRIE_MASK;
204  }
205  }
207 
208  protected:
210  // Listeners of events on the buffer
211  virtual void on_put() override
212  {
213  errors_.all_errors.queue_overflow = ostreambuf::overflow();
214  synchronized
215  {
216  // Check if TX is not currently active, if so, activate it
217  if (!transmitting_)
218  {
219  // Yes, trigger TX
220  char value;
221  if (queue().pull_(value))
222  {
223  // Set UDR interrupt to be notified when we can send the next character
224  TRAIT::UCSRB |= TRAIT::UDRIE_MASK;
225  TRAIT::UDR = value;
226  transmitting_ = true;
227  }
228  }
229  }
230  }
232 
233  private:
234  bool transmitting_;
235  };

320  inline void data_receive_complete()
321  {
322  char status = TRAIT::UCSRA;
323  errors_.all_errors.data_overrun = status & TRAIT::DOR_MASK;
324  errors_.all_errors.frame_error = status & TRAIT::FE_MASK;
325  errors_.all_errors.parity_error = status & TRAIT::UPE_MASK;
326  char value = TRAIT::UDR;
327  errors_.all_errors.queue_overflow = !queue().push_(value);
